```typescript
/**
 * Phase 19 — D-PRIV-3 + AI-SPEC §7 production monitoring telemetry.
 *
 * Architecture-purity (extends Plan 01 grep guards): zero @anthropic-ai/sdk,
 * @modelcontextprotocol/sdk, better-sqlite3, drizzle-orm imports. Pure helper
 * emitting structured log lines via console.error (matches Phase 14 c2pa-config
 * boot-log convention).
 *
 * D-PRIV-3 strict contract: counts + timings ONLY. NEVER include prompt text,
 * response text, summary_text, prompt_positive, prompt_negative, user_prompt
 * fields. Multi-encoding leak scan applies to every emit (extends v1.1
 * cross-cutting invariant — RESEARCH.md "Multi-encoding leak scan" thread).
 *
 * Sampling per AI-SPEC §7 Smart Sampling Strategy:
 *   - Every fallback event: full log
 *   - Every live event: full metadata
 *   - 1% sample of cache_hit (deterministic by hash(version_id + timestamp_minute) % 100 == 0)
 *
 * The flattenAnthropicError helper (Plan 04) is the upstream defence on error
 * paths — telemetry NEVER receives raw error.message strings. The reason field
 * is a structured outcome enum (e.g., 'circuit_open', 'validation_failed'),
 * not a free-form error message.
 */

import { flattenAnthropicError } from './anthropic-client.js';

export type SummaryTelemetryEvent = {
  event: 'summary_generated';
  version_id: string;
  manifest_sha256: string | null;
  model_id: string;
  template_version: string;
  duration_ms: number;
  prompt_tokens: number;
  completion_tokens: number;
  outcome: 'cache_hit' | 'live' | 'fallback';
  reason?: string;
};

/** Banned field names — defence-in-depth check before emit. */
const BANNED_FIELDS = [
  'text', 'summary_text', 'prompt_positive', 'prompt_negative',
  'user_prompt', 'system_prompt', 'response', 'response_text',
] as const;

/**
 * Deterministic 1% sampling for cache_hit — uses a simple hash on
 * (version_id, current minute bucket) so a given version's cache_hit emits
 * at a stable 1% rate without needing crypto.random.
 */
function shouldSampleCacheHit(versionId: string, timestampMs: number): boolean {
  const minute = Math.floor(timestampMs / 60_000);
  let hash = 0;
  const str = `${versionId}:${minute}`;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return Math.abs(hash) % 100 === 0;
}

/**
 * Defence-in-depth — verify the event object does NOT contain any banned
 * field name. Throws Error if it does (caught by caller; logged via
 * flattenAnthropicError-scrubbed message).
 */
function assertNoBannedFields(event: Record<string, unknown>): void {
  for (const key of Object.keys(event)) {
    if (BANNED_FIELDS.includes(key as typeof BANNED_FIELDS[number])) {
      throw new Error(`telemetry assertion: banned field "${key}" in summary event payload`);
    }
  }
}

/**
 * Emit a structured telemetry event. Counts + timings only per D-PRIV-3.
 *
 * Sampling rules (per AI-SPEC §7 Smart Sampling):
 *   outcome='fallback' → always emit
 *   outcome='live'     → always emit
 *   outcome='cache_hit' → emit on 1% deterministic sample
 */
export function logSummaryEvent(event: SummaryTelemetryEvent, clock: () => number = Date.now): void {
  // Defence-in-depth — assert no banned fields snuck into the payload.
  assertNoBannedFields(event as unknown as Record<string, unknown>);

  // Sampling decision.
  if (event.outcome === 'cache_hit') {
    if (!shouldSampleCacheHit(event.version_id, clock())) return;
  }

  // Multi-encoding leak scan on the JSON-serialized payload (defence-in-depth).
  const payload = JSON.stringify(event);
  const apiKey = process.env.ANTHROPIC_API_KEY ?? '';
  if (apiKey.length > 0) {
    const fragments = [
      apiKey,
      Buffer.from(apiKey, 'utf16le').toString('binary'),
      Buffer.from(apiKey, 'utf16le').reverse().toString('binary'),
      Buffer.from(apiKey).toString('base64'),
    ];
    for (const frag of fragments) {
      if (frag.length === 0) continue;
      if (payload.includes(frag)) {
        // Refuse to emit — this is a critical contract failure.
        console.error('[summary-telemetry] EMIT REFUSED — API key fragment in payload');
        return;
      }
    }
  }

  // Emit to console.error (matches Phase 14 c2pa-config boot-log convention).
  // The vfx-familiar prefix marks this as engine-layer telemetry.
  console.error(`vfx-familiar: ${payload}`);
}

/**
 * Test-only — re-exported flattenAnthropicError for telemetry test verification.
 * Production code should NOT call this directly; the engine facade composes it.
 */
export { flattenAnthropicError };
```
