```typescript
// Phase 19 — SUM-05/06. Dashboard-side helpers for the AI conversational
// summary feature. NEVER throw — defensive error-collapse to a typed error
// state, mirroring getC2paStatus pattern at lines 332-354.

export type SummaryFetchResponse =
  | {
      state: 'success';
      text: string;
      source: 'live' | 'cache_hit';
      generated_at: string;
      template_version: string;
      model_id: string;
      regenerateAvailableAtMs: number | null;
    }
  | {
      state: 'fallback';
      text: string;
      source: 'fallback';
      reason?: string;
      regenerateAvailableAtMs: number | null;
    }
  | { state: 'error'; message?: string };

/**
 * GET /api/versions/:id/summary
 *
 * Maps the server's SummaryOutcome envelope (cache_hit | live | fallback)
 * into the dashboard's SummaryFetchResponse shape. Defensive — collapses
 * network errors / parse errors / unexpected envelopes to
 * { state: 'error' } per Phase 14 getC2paStatus precedent.
 */
export async function getSummary(versionId: string): Promise<SummaryFetchResponse> {
  try {
    const res = await fetch(`/api/versions/${encodeURIComponent(versionId)}/summary`);
    if (!res.ok) {
      return { state: 'error', message: `HTTP ${res.status}` };
    }
    const data = await res.json();
    return mapServerEnvelope(data);
  } catch (err) {
    return { state: 'error', message: err instanceof Error ? err.message : 'unknown' };
  }
}

/**
 * POST /api/versions/:id/summary/regenerate
 *
 * Forces a fresh LLM call (server bypasses cache lookup). Returns the same
 * SummaryFetchResponse shape. Defensive — 429 throttle response collapses
 * to { state: 'error' } and the caller's previous summary state stays
 * visible (the 60s cooldown countdown handles the visual feedback).
 */
export async function regenerateSummary(versionId: string): Promise<SummaryFetchResponse> {
  try {
    const res = await fetch(`/api/versions/${encodeURIComponent(versionId)}/summary/regenerate`, {
      method: 'POST',
    });
    if (!res.ok) {
      // 429 throttle, 5xx server error, etc. — surface as error envelope.
      // The dashboard countdown timer uses the regenerate_available_at_ms
      // from the previous successful fetch (or undefined → button enabled).
      return { state: 'error', message: `HTTP ${res.status}` };
    }
    const data = await res.json();
    return mapServerEnvelope(data);
  } catch (err) {
    return { state: 'error', message: err instanceof Error ? err.message : 'unknown' };
  }
}

/**
 * Map the server's SummaryOutcome envelope shape (with regenerate_available_at_ms
 * augmentation) to the dashboard's SummaryFetchResponse shape.
 */
function mapServerEnvelope(data: unknown): SummaryFetchResponse {
  if (typeof data !== 'object' || data === null) {
    return { state: 'error', message: 'malformed response' };
  }
  const envelope = data as Record<string, unknown>;
  const source = envelope.source;
  const text = typeof envelope.text === 'string' ? envelope.text : '';
  const regenerateAvailableAtMs = typeof envelope.regenerate_available_at_ms === 'number'
    ? envelope.regenerate_available_at_ms
    : null;

  if (source === 'cache_hit' || source === 'live') {
    return {
      state: 'success',
      text,
      source,
      generated_at: typeof envelope.generated_at === 'string' ? envelope.generated_at : '',
      template_version: typeof envelope.template_version === 'string' ? envelope.template_version : '',
      model_id: typeof envelope.model_id === 'string' ? envelope.model_id : '',
      regenerateAvailableAtMs,
    };
  }
  if (source === 'fallback') {
    return {
      state: 'fallback',
      text,
      source: 'fallback',
      reason: typeof envelope.reason === 'string' ? envelope.reason : undefined,
      regenerateAvailableAtMs,
    };
  }
  return { state: 'error', message: 'unexpected source' };
}
```

export type SummaryState =
  | { state: 'loading' }
  | {
      state: 'success';
      text: string;
      source: 'live' | 'cache_hit';
      generated_at: string;
      template_version: string;
      model_id: string;
      regenerateAvailableAtMs: number | null;
    }
  | {
      state: 'fallback';
      text: string;
      source: 'fallback';
      reason?: string;
      regenerateAvailableAtMs: number | null;
    }
  | { state: 'error'; message?: string };

/**
 * Per-version summary state. Read by SummarySection (Plan 06); written by
 * VersionDrawer's auto-fetch effect (Plan 06).
 */
export const summarySignal = signal<Map<string, SummaryState>>(new Map());

/**
 * fetchSummary helper used by VersionDrawer's useEffect mount + Regenerate
 * handler. Wraps lib/api.getSummary / regenerateSummary; NEVER throws —
 * collapses errors to { state: 'error' } per defensive contract.
 *
 * The caller is responsible for writing the result back into summarySignal:
 *   const state = await fetchSummary(versionId);
 *   summarySignal.value = new Map(summarySignal.value).set(versionId, state);
 */
export async function fetchSummary(
  versionId: string,
  options: { regenerate?: boolean } = {},
): Promise<SummaryState> {
  const response: SummaryFetchResponse = options.regenerate
    ? await regenerateSummary(versionId)
    : await getSummary(versionId);
  return mapResponseToState(response);
}

function mapResponseToState(response: SummaryFetchResponse): SummaryState {
  if (response.state === 'success') {
    return {
      state: 'success',
      text: response.text,
      source: response.source,
      generated_at: response.generated_at,
      template_version: response.template_version,
      model_id: response.model_id,
      regenerateAvailableAtMs: response.regenerateAvailableAtMs,
    };
  }
  if (response.state === 'fallback') {
    return {
      state: 'fallback',
      text: response.text,
      source: 'fallback',
      reason: response.reason,
      regenerateAvailableAtMs: response.regenerateAvailableAtMs,
    };
  }
  return { state: 'error', message: response.message };
}
