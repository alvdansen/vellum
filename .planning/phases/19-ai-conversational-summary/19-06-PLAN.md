**Part B — Create `packages/dashboard/src/components/RegenerateButton.tsx`**:

```tsx
/**
 * RegenerateButton — Phase 19 SUM-04 thin presentational button + cooldown timer.
 *
 * Architecture-purity (D-WEBUI-31): zero imports from src/. Pure component
 * with @preact/hooks for the 1Hz countdown tick.
 *
 * Per UI-SPEC §"<RegenerateButton/> API contract": three render states.
 * The 1Hz timer is acceptable per prefers-reduced-motion: reduce because
 * changing TEXT (not animation) does not violate the reduced-motion contract.
 *
 * Accessibility: native <button type="button"> + aria-label + aria-busy
 * + native disabled attribute (HTML disabled removes from tab order).
 */

import { useEffect, useState } from 'preact/hooks';
import {
  REGENERATE_BUTTON_LABEL,
  REGENERATE_BUTTON_FETCHING,
  regenerateButtonCooldownLabel,
} from '../lib/copy.js';

export interface RegenerateButtonProps {
  /**
   * Server-reported epoch ms at which the next regenerate call will succeed.
   * When > Date.now(), the button is disabled and the label shows the
   * countdown. When null/undefined, the button is enabled by default.
   */
  regenerateAvailableAtMs?: number | null;
  /** True while a fetch is in flight (initial mount OR active regenerate). */
  isFetching?: boolean;
  /** Click handler — parent applies 500ms debounce + invokes the actual fetch. */
  onClick: () => void;
  /** Required ARIA label — caller passes regenerateButtonAriaLabel(version.label). */
  ariaLabel: string;
  /** Optional class for composition with parent. */
  class?: string;
}

export function RegenerateButton({
  regenerateAvailableAtMs,
  isFetching = false,
  onClick,
  ariaLabel,
  class: className,
}: RegenerateButtonProps) {
  // 1Hz tick for the cooldown countdown digit.
  const [now, setNow] = useState(() => Date.now());
  useEffect(() => {
    if (!regenerateAvailableAtMs || regenerateAvailableAtMs <= now) return;
    const id = setInterval(() => setNow(Date.now()), 1000);
    return () => clearInterval(id);
  }, [regenerateAvailableAtMs, now]);

  const cooldownSeconds = regenerateAvailableAtMs && regenerateAvailableAtMs > now
    ? Math.ceil((regenerateAvailableAtMs - now) / 1000)
    : 0;

  const isDisabled = isFetching || cooldownSeconds > 0;
  const label = isFetching
    ? REGENERATE_BUTTON_FETCHING
    : cooldownSeconds > 0
    ? regenerateButtonCooldownLabel(cooldownSeconds)
    : REGENERATE_BUTTON_LABEL;

  // Match View Diff button styling verbatim per UI-SPEC.
  const baseClass = 'rounded bg-[var(--color-accent)] px-2 py-1 text-xs font-normal text-[var(--color-bg)] hover:opacity-90 disabled:opacity-50';
  const composed = className ? `${baseClass} ${className}` : baseClass;

  return (
    <button
      type="button"
      class={composed}
      aria-label={ariaLabel}
      aria-busy={isFetching ? 'true' : 'false'}
      disabled={isDisabled}
      onClick={isDisabled ? undefined : onClick}
    >
      {label}
    </button>
  );
}
```

**Surgical change 2 — Add summary state hooks + auto-fetch effect + Regenerate handler** (alongside existing C2paStatus hook block):

```tsx
// Phase 19 — SUM-01/04/06. Per-version summary state + auto-fetch + 500ms
// debounce on Regenerate. Mirrors the C2PA-status pattern at lines 119-133
// verbatim (let alive = true cancellation; defensive .catch).
const [summary, setSummary] = useState<SummaryState>({ state: 'loading' });

// D-PRIV-2 first-use disclosure — read localStorage at mount; auto-ack on
// first regenerate click (UI-SPEC researcher recommendation).
const [showFirstUseDisclosure, setShowFirstUseDisclosure] = useState<boolean>(() => {
  try {
    return typeof localStorage !== 'undefined'
      && localStorage.getItem(SUMMARY_FIRST_USE_LOCALSTORAGE_KEY) !== 'true';
  } catch { return false; }  // Defensive — SSR / privacy-mode browsers
});

// 500ms client-side debounce for Regenerate (SUM-04 + D-FB-4).
const lastRegenerateClickRef = useRef<number>(0);

// Auto-fetch summary on version.id change. Mirrors C2PA effect at lines 119-133.
useEffect(() => {
  let alive = true;
  setSummary({ state: 'loading' });
  fetchSummary(version.id)
    .then((s) => {
      if (alive) {
        setSummary(s);
        // Mirror to the global signal for cross-component reads (Plan 05).
        summarySignal.value = new Map(summarySignal.value).set(version.id, s);
      }
    })
    .catch(() => {
      if (alive) setSummary({ state: 'error' });
    });
  return () => { alive = false; };
}, [version.id]);

async function handleRegenerate(): Promise<void> {
  // 500ms client-side debounce — captures the most-recent click; ignores rapid succession.
  const now = Date.now();
  if (now - lastRegenerateClickRef.current < 500) return;
  lastRegenerateClickRef.current = now;

  // D-PRIV-2 — auto-ack first-use disclosure on first regenerate click.
  if (showFirstUseDisclosure) {
    try {
      localStorage.setItem(SUMMARY_FIRST_USE_LOCALSTORAGE_KEY, 'true');
    } catch { /* defensive — privacy-mode or SSR */ }
    setShowFirstUseDisclosure(false);
  }

  // Optimistic UI flip.
  setSummary({ state: 'loading' });
  try {
    const fresh = await fetchSummary(version.id, { regenerate: true });
    setSummary(fresh);
    summarySignal.value = new Map(summarySignal.value).set(version.id, fresh);
  } catch {
    setSummary({ state: 'error' });
  }
}

// Read regenerateAvailableAtMs from the current summary state (only present on
// success/fallback variants — error/loading state defaults to undefined).
const regenerateAvailableAtMs =
  summary.state === 'success' || summary.state === 'fallback'
    ? summary.regenerateAvailableAtMs
    : null;
```
