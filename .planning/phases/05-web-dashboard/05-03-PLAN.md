```
*_NOT_FOUND                 → 404
PROVENANCE_UNAVAILABLE      → 404  (explicit — does not match _NOT_FOUND)
OUTPUT_UNAVAILABLE          → 404  (explicit — does not match _NOT_FOUND)
INVALID_*                   → 400  (matches: INVALID_INPUT, INVALID_SHOT_FORMAT, INVALID_WORKFLOW_FORMAT, INVALID_SCOPE, ITERATE_INVALID_PATCH)

REPRODUCE_BLOCKED           → 422
```

```typescript
import type { ErrorHandler } from 'hono';
import { TypedError } from '../engine/errors.js';

// Explicit 404 set — includes *_NOT_FOUND pattern plus *_UNAVAILABLE codes
const NOT_FOUND_CODES = new Set([
  'WORKSPACE_NOT_FOUND',
  'PROJECT_NOT_FOUND',
  'SEQUENCE_NOT_FOUND',
  'SHOT_NOT_FOUND',
  'VERSION_NOT_FOUND',
  'PARENT_NOT_FOUND',
  'PROVENANCE_UNAVAILABLE',
  'OUTPUT_UNAVAILABLE',
]);

// Explicit 422 set — unprocessable / pre-condition failures
const UNPROCESSABLE_CODES = new Set([
  'REPRODUCE_BLOCKED',
  'VERSION_NOT_COMPLETED',
  'TAG_LIMIT_EXCEEDED',
  'METADATA_LIMIT_EXCEEDED',
]);

// Explicit 409 set — conflict
const CONFLICT_CODES = new Set([
  'DUPLICATE_NAME',
  'CONCURRENT_SUBMIT_CONFLICT',
]);

// Explicit 502 set — gateway / external service failures
const BAD_GATEWAY_CODES = new Set([
  'COMFYUI_API_ERROR',
  'COMFYUI_CREDENTIALS_MISSING',
  'COMFYUI_RATE_LIMITED',
  'GENERATION_TIMEOUT',
  'DOWNLOAD_FAILED',
]);

export function statusForCode(code: string): number {
  if (NOT_FOUND_CODES.has(code)) return 404;
  if (code.startsWith('INVALID_') || code === 'TAG_INVALID' || code === 'METADATA_INVALID') return 400;
  if (BAD_GATEWAY_CODES.has(code) || code.startsWith('COMFYUI_')) return 502;
  if (UNPROCESSABLE_CODES.has(code)) return 422;
  if (CONFLICT_CODES.has(code)) return 409;
  return 500;
}

export const typedErrorHandler: ErrorHandler = (err, c) => {
  if (err instanceof TypedError) {
    return c.json(
      { error: { code: err.code, message: err.message } },
      statusForCode(err.code) as Parameters<typeof c.json>[1],
    );
  }
  return c.json(
    { error: { code: 'INTERNAL_ERROR', message: err.message } },
    500,
  );
};
```
