NOTE — `GET /api/versions/:id/output` is NOT an Engine method call.
It streams from disk using `fs.createReadStream` from `outputs/<versionId>/<filename>`.
The filename comes from the version's `outputs_json` field (call `engine.getVersion(id)` first to
get the filename, then stream). Per D-WEBUI-26 + D-WEBUI-33: path is `outputs/<version_id>/<filename>`.
The route must validate the filename (no `..`, no `/`) before constructing the fs path.

Output file path construction:
```typescript
import { createReadStream, existsSync } from 'node:fs';
import { Readable } from 'node:stream';
import * as path from 'node:path';

// In route handler for GET /api/versions/:id/output:
const version = engine.getVersion(c.req.param('id'));
const outputs = version.entity.outputs_json
  ? (JSON.parse(version.entity.outputs_json as string) as string[])
  : [];
if (outputs.length === 0) throw new TypedError('OUTPUT_UNAVAILABLE', 'No outputs for this version');
const filename = path.basename(outputs[0]); // first output file
if (!filename || filename.includes('..')) throw new TypedError('INVALID_INPUT', 'Invalid output path');
const ext = path.extname(filename).toLowerCase();
const mime = MIME_MAP[ext] ?? 'application/octet-stream';
const filePath = path.join('outputs', c.req.param('id'), filename);
if (!existsSync(filePath)) throw new TypedError('OUTPUT_UNAVAILABLE', 'Output file not on disk');
const stream = createReadStream(filePath);
const webStream = Readable.toWeb(stream) as ReadableStream;
return c.body(webStream, 200, { 'Content-Type': mime });
```

```typescript
import { Hono } from 'hono';
import { createReadStream, existsSync } from 'node:fs';
import { Readable } from 'node:stream';
import * as path from 'node:path';
import type { Engine } from '../engine/pipeline.js';
import { TypedError } from '../engine/errors.js';
import { typedErrorHandler } from './error-middleware.js';

const MIME_MAP: Record<string, string> = {
  '.png': 'image/png',
  '.jpg': 'image/jpeg',
  '.jpeg': 'image/jpeg',
  '.webp': 'image/webp',
  '.mp4': 'video/mp4',
  '.gif': 'image/gif',
};

export function createDashboardRouter(engine: Engine) {
  const app = new Hono();
  app.onError(typedErrorHandler);

  // --- Workspaces ---
  app.get('/api/workspaces', (c) => {
    const limit = Number(c.req.query('limit') ?? 20);
    const offset = Number(c.req.query('offset') ?? 0);
    return c.json(engine.listWorkspaces(limit, offset));
  });
  app.get('/api/workspaces/:id', (c) =>
    c.json(engine.getWorkspace(c.req.param('id'))));
  app.get('/api/workspaces/:id/projects', (c) => {
    const limit = Number(c.req.query('limit') ?? 20);
    const offset = Number(c.req.query('offset') ?? 0);
    return c.json(engine.listProjects(c.req.param('id'), limit, offset));
  });

  // --- Projects ---
  app.get('/api/projects/:id', (c) =>
    c.json(engine.getProject(c.req.param('id'))));
  app.get('/api/projects/:id/sequences', (c) => {
    const limit = Number(c.req.query('limit') ?? 20);
    const offset = Number(c.req.query('offset') ?? 0);
    return c.json(engine.listSequences(c.req.param('id'), limit, offset));
  });

  // --- Sequences ---
  app.get('/api/sequences/:id', (c) =>
    c.json(engine.getSequence(c.req.param('id'))));
  app.get('/api/sequences/:id/shots', (c) => {
    const limit = Number(c.req.query('limit') ?? 20);
    const offset = Number(c.req.query('offset') ?? 0);
    return c.json(engine.listShots(c.req.param('id'), limit, offset));
  });

  // --- Shots ---
  app.get('/api/shots/:id', (c) =>
    c.json(engine.getShot(c.req.param('id'))));
  app.get('/api/shots/:id/versions', (c) => {
    const limit = Number(c.req.query('limit') ?? 20);
    const offset = Number(c.req.query('offset') ?? 0);
    const include_tags = c.req.query('include_tags') === 'true';
    const include_metadata = c.req.query('include_metadata') === 'true';
    return c.json(engine.listVersionsForShot(c.req.param('id'), limit, offset, { include_tags, include_metadata }));
  });

  // --- Versions ---
  app.get('/api/versions/:id', (c) =>
    c.json(engine.getVersion(c.req.param('id'))));
  app.get('/api/versions/:id/provenance', (c) =>
    c.json(engine.getProvenance(c.req.param('id'))));
  app.get('/api/versions/:id/diff', (c) => {
    const against = c.req.query('against');
    if (!against) throw new TypedError('INVALID_INPUT', 'Missing required query param: against');
    return c.json(engine.diffVersions(c.req.param('id'), against));
  });

  // --- Output streaming (D-WEBUI-26 + D-WEBUI-33) ---
  // NOT an Engine method — streams from disk at outputs/<versionId>/<filename>
  // Validate filename from outputs_json before constructing fs path (T-5-04)
  app.get('/api/versions/:id/output', (c) => {
    const versionId = c.req.param('id');
    const version = engine.getVersion(versionId);
    const outputsRaw = version.entity.outputs_json;
    const outputs: string[] = outputsRaw
      ? (JSON.parse(outputsRaw as string) as string[])
      : [];
    if (outputs.length === 0) {
      throw new TypedError('OUTPUT_UNAVAILABLE', 'No outputs for this version');
    }
    const filename = path.basename(outputs[0]);
    if (!filename || filename.includes('..') || filename.includes('/') || filename.includes('\\')) {
      throw new TypedError('INVALID_INPUT', 'Invalid output path');
    }
    const ext = path.extname(filename).toLowerCase();
    const mime = MIME_MAP[ext] ?? 'application/octet-stream';
    const filePath = path.join('outputs', versionId, filename);
    if (!existsSync(filePath)) {
      throw new TypedError('OUTPUT_UNAVAILABLE', 'Output file not found on disk');
    }
    const stream = createReadStream(filePath);
    const webStream = Readable.toWeb(stream) as ReadableStream;
    return c.body(webStream, 200, { 'Content-Type': mime });
  });

  // --- Reproduce ---
  app.post('/api/versions/:id/reproduce', async (c) => {
    const body = await c.req.json().catch(() => ({}));
    const result = await engine.reproduceVersion(c.req.param('id'), body?.notes);
    return c.json(result, 201);
  });

  // --- Assets ---
  app.post('/api/assets/query', async (c) => {
    const body = await c.req.json();
    return c.json(engine.queryAssets(body));
  });
  app.post('/api/assets/list_tags', async (c) => {
    const body = await c.req.json();
    return c.json(engine.listTags(body));
  });
  app.post('/api/assets/list_metadata_keys', async (c) => {
    const body = await c.req.json();
    return c.json(engine.listMetadataKeys(body));
  });

  // --- Dashboard home ---
  app.get('/api/dashboard/home', (c) =>
    c.json(engine.getDashboardHome()));

  return app;
}
```
