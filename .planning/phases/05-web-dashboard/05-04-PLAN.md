```typescript
import { Hono } from 'hono';
import { createReadStream } from 'node:fs';
import { Readable } from 'node:stream';
import * as path from 'node:path';
import type { Engine } from '../engine/pipeline.js';
import { TypedError } from '../engine/errors.js';
import { typedErrorHandler } from './error-middleware.js';

const MIME_MAP: Record<string, string> = {
  '.png': 'image/png',
  '.jpg': 'image/jpeg',
  '.webp': 'image/webp',
  '.mp4': 'video/mp4',
};

export function createDashboardRouter(engine: Engine) {
  const app = new Hono();
  app.onError(typedErrorHandler);

  // --- Workspaces ---
  app.get('/api/workspaces', async (c) => c.json(await engine.listWorkspaces()));
  app.get('/api/workspaces/:id', async (c) => c.json(await engine.getWorkspace(c.req.param('id'))));
  app.get('/api/workspaces/:id/projects', async (c) =>
    c.json(await engine.listProjects(c.req.param('id'))));

  // --- Projects ---
  app.get('/api/projects/:id', async (c) => c.json(await engine.getProject(c.req.param('id'))));
  app.get('/api/projects/:id/sequences', async (c) =>
    c.json(await engine.listSequences(c.req.param('id'))));

  // --- Sequences ---
  app.get('/api/sequences/:id', async (c) => c.json(await engine.getSequence(c.req.param('id'))));
  app.get('/api/sequences/:id/shots', async (c) =>
    c.json(await engine.listShots(c.req.param('id'))));

  // --- Shots ---
  app.get('/api/shots/:id', async (c) => c.json(await engine.getShot(c.req.param('id'))));
  app.get('/api/shots/:id/versions', async (c) =>
    c.json(await engine.listVersions(c.req.param('id'))));

  // --- Versions ---
  app.get('/api/versions/:id', async (c) => c.json(await engine.getVersion(c.req.param('id'))));
  app.get('/api/versions/:id/provenance', async (c) =>
    c.json(await engine.getProvenance(c.req.param('id'))));
  app.get('/api/versions/:id/diff', async (c) =>
    c.json(await engine.getDiff(c.req.param('id'))));
  app.get('/api/versions/:id/outputs', async (c) =>
    c.json(await engine.listOutputs(c.req.param('id'))));

  // --- Output file streaming (T-5-04: validate filename before fs access) ---
  app.get('/api/outputs/:versionId/:filename', async (c) => {
    const { versionId, filename } = c.req.param();
    if (filename.includes('..') || filename.includes('/') || filename.includes('\\')) {
      throw new TypedError('INVALID_INPUT', 'Invalid filename');
    }
    const ext = path.extname(filename).toLowerCase();
    const mime = MIME_MAP[ext] ?? 'application/octet-stream';
    const filePath = path.join(engine.outputsDir, versionId, filename);
    const stream = createReadStream(filePath);
    const webStream = Readable.toWeb(stream) as ReadableStream;
    return c.body(webStream, 200, { 'Content-Type': mime });
  });

  // --- Tags ---
  app.get('/api/tags', async (c) => c.json(await engine.listTags()));
  app.get('/api/tags/:id', async (c) => c.json(await engine.getTag(c.req.param('id'))));
  app.patch('/api/tags/:id', async (c) => {
    const patch = await c.req.json();
    return c.json(await engine.updateTag(c.req.param('id'), patch));
  });
  app.delete('/api/tags/:id', async (c) => {
    await engine.deleteTag(c.req.param('id'));
    return c.body(null, 204);
  });

  return app;
}
```
