```typescript
import { streamSSE } from 'hono/streaming';
import type { Context } from 'hono';
import type { Engine } from '../engine/pipeline.js';
import type { EngineEventMap } from '../engine/events.js';

const EVENT_TYPES = [
  'version.status_changed',
  'version.created',
  'tag.changed',
  'metadata.changed',
  'hierarchy.created',
] as const satisfies ReadonlyArray<keyof EngineEventMap>;

export function createSseHandler(engine: Engine, allowedOrigins: string[] = []) {
  return (c: Context) => {
    // Origin check (T-5-01)
    const origin = c.req.header('Origin') ?? '';
    if (allowedOrigins.length > 0 && !allowedOrigins.includes(origin)) {
      return c.text('Forbidden', 403);
    }

    return streamSSE(c, async (stream) => {
      // Register listeners for all 5 event types
      const listeners = {} as Record<keyof EngineEventMap, (payload: unknown) => void>;

      for (const type of EVENT_TYPES) {
        const listener = (payload: unknown) => {
          stream.writeSSE({ data: JSON.stringify(payload), event: type });
        };
        listeners[type] = listener as (payload: EngineEventMap[typeof type]) => void;
        engine.events.onEvent(type, listener as (payload: EngineEventMap[typeof type]) => void);
      }

      // Keep-alive every 30s (T-5-08: prevents proxy from closing idle connections)
      const keepAliveInterval = setInterval(() => {
        stream.writeSSE({ data: ': ping' });
      }, 30_000);

      // Cleanup on disconnect
      const cleanup = () => {
        clearInterval(keepAliveInterval);
        for (const type of EVENT_TYPES) {
          engine.events.offEvent(
            type,
            listeners[type] as (payload: EngineEventMap[typeof type]) => void,
          );
        }
      };

      const signal = c.req.raw.signal;
      signal.addEventListener('abort', cleanup);

      // Hold the stream open until client disconnects
      await new Promise<void>((resolve) => {
        signal.addEventListener('abort', resolve);
      });

      cleanup();
    });
  };
}
```
