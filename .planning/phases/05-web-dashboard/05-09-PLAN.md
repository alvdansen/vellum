Create `packages/dashboard/src/components/ThemeToggle.tsx`:

```tsx
import { h } from 'preact';
import { useState, useEffect } from 'preact/hooks';
import { Sun, Moon } from 'lucide-preact';

const STORAGE_KEY = 'vfx-familiar:theme';

export function ThemeToggle() {
  const [theme, setTheme] = useState<'dark' | 'light'>('dark');

  useEffect(() => {
    const saved = localStorage.getItem(STORAGE_KEY) as 'dark' | 'light' | null;
    const initial = saved ?? 'dark';
    setTheme(initial);
    document.documentElement.setAttribute('data-theme', initial);
  }, []);

  function toggle() {
    const next = theme === 'dark' ? 'light' : 'dark';
    setTheme(next);
    localStorage.setItem(STORAGE_KEY, next);
    document.documentElement.setAttribute('data-theme', next);
  }

  return (
    <button
      onClick={toggle}
      aria-label={`Switch to ${theme === 'dark' ? 'light' : 'dark'} mode`}
      class="rounded p-1.5 text-[var(--color-text-secondary)] hover:text-[var(--color-text-primary)] hover:bg-[var(--color-surface)]"
    >
      {theme === 'dark' ? <Sun size={16} /> : <Moon size={16} />}
    </button>
  );
}
```

```tsx
import { h, Fragment } from 'preact';
import { ChevronRight, ChevronDown } from 'lucide-preact';
import type { Workspace, Project, Sequence, Shot } from '../types/entities.js';

interface TreeSidebarProps {
  workspaces: Workspace[];
  selectedShotId: string | null;
  onSelectShot: (shotId: string) => void;
  expandedIds: Set<string>;
  onToggleExpand: (id: string) => void;
}

export function TreeSidebar({
  workspaces,
  selectedShotId,
  onSelectShot,
  expandedIds,
  onToggleExpand,
}: TreeSidebarProps) {
  return (
    <nav
      class="flex flex-col gap-0.5 overflow-y-auto"
      aria-label="Project hierarchy"
      style={{ width: 'var(--sidebar-width)' }}
    >
      {workspaces.map((ws) => (
        <WorkspaceNode
          key={ws.id}
          workspace={ws}
          selectedShotId={selectedShotId}
          onSelectShot={onSelectShot}
          expandedIds={expandedIds}
          onToggleExpand={onToggleExpand}
        />
      ))}
    </nav>
  );
}

function WorkspaceNode({ workspace, selectedShotId, onSelectShot, expandedIds, onToggleExpand }: {
  workspace: Workspace & { projects?: Project[] };
  selectedShotId: string | null;
  onSelectShot: (id: string) => void;
  expandedIds: Set<string>;
  onToggleExpand: (id: string) => void;
}) {
  const expanded = expandedIds.has(workspace.id);
  return (
    <>
      <TreeRow
        label={workspace.name}
        depth={0}
        expanded={expanded}
        hasChildren={!!workspace.projects?.length}
        onToggle={() => onToggleExpand(workspace.id)}
        isSelected={false}
        onClick={() => onToggleExpand(workspace.id)}
      />
      {expanded && workspace.projects?.map((proj) => (
        <ProjectNode
          key={proj.id}
          project={proj}
          selectedShotId={selectedShotId}
          onSelectShot={onSelectShot}
          expandedIds={expandedIds}
          onToggleExpand={onToggleExpand}
        />
      ))}
    </>
  );
}

function ProjectNode({ project, selectedShotId, onSelectShot, expandedIds, onToggleExpand }: {
  project: Project & { sequences?: Sequence[] };
  selectedShotId: string | null;
  onSelectShot: (id: string) => void;
  expandedIds: Set<string>;
  onToggleExpand: (id: string) => void;
}) {
  const expanded = expandedIds.has(project.id);
  return (
    <>
      <TreeRow
        label={project.name}
        depth={1}
        expanded={expanded}
        hasChildren={!!project.sequences?.length}
        onToggle={() => onToggleExpand(project.id)}
        isSelected={false}
        onClick={() => onToggleExpand(project.id)}
      />
      {expanded && project.sequences?.map((seq) => (
        <SequenceNode
          key={seq.id}
          sequence={seq}
          selectedShotId={selectedShotId}
          onSelectShot={onSelectShot}
          expandedIds={expandedIds}
          onToggleExpand={onToggleExpand}
        />
      ))}
    </>
  );
}

function SequenceNode({ sequence, selectedShotId, onSelectShot, expandedIds, onToggleExpand }: {
  sequence: Sequence & { shots?: Shot[] };
  selectedShotId: string | null;
  onSelectShot: (id: string) => void;
  expandedIds: Set<string>;
  onToggleExpand: (id: string) => void;
}) {
  const expanded = expandedIds.has(sequence.id);
  return (
    <>
      <TreeRow
        label={sequence.name}
        depth={2}
        expanded={expanded}
        hasChildren={!!sequence.shots?.length}
        onToggle={() => onToggleExpand(sequence.id)}
        isSelected={false}
        onClick={() => onToggleExpand(sequence.id)}
      />
      {expanded && sequence.shots?.map((shot) => (
        <TreeRow
          key={shot.id}
          label={shot.name}
          depth={3}
          expanded={false}
          hasChildren={false}
          isSelected={shot.id === selectedShotId}
          onClick={() => onSelectShot(shot.id)}
          onToggle={() => {}}
        />
      ))}
    </>
  );
}

function TreeRow({
  label, depth, expanded, hasChildren, isSelected, onClick, onToggle,
}: {
  label: string;
  depth: number;
  expanded: boolean;
  hasChildren: boolean;
  isSelected: boolean;
  onClick: () => void;
  onToggle: () => void;
}) {
  const Icon = expanded ? ChevronDown : ChevronRight;
  return (
    <div
      class={`flex cursor-pointer items-center gap-1 rounded px-2 py-1 text-sm transition-colors ${
        isSelected
          ? 'bg-[var(--color-accent)] text-[var(--color-bg)]'
          : 'text-[var(--color-text-primary)] hover:bg-[var(--color-surface)]'
      }`}
      style={{ paddingLeft: `${8 + depth * 12}px` }}
      onClick={onClick}
      role="treeitem"
      aria-expanded={hasChildren ? expanded : undefined}
    >
      {hasChildren ? (
        <span
          class="flex-shrink-0"
          onClick={(e) => { e.stopPropagation(); onToggle(); }}
        >
          <Icon size={14} />
        </span>
      ) : (
        <span class="w-3.5 flex-shrink-0" />
      )}
      <span class="truncate">{label}</span>
    </div>
  );
}
```
