import { serveStatic } from '@hono/node-server/serve-static';
import { existsSync } from 'node:fs';
import * as path from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const distPath = path.resolve(__dirname, '../../packages/dashboard/dist');

export function createStaticHandler() {
  return async (c: Context, next: () => Promise<void>) => {
    if (!existsSync(distPath)) {
      // Fallback HTML — dist not yet built
      return c.html(`<!DOCTYPE html>
<html><head><title>VFX Familiar</title></head>
<body><p>Dashboard not built. Run: npm run build:dashboard</p></body></html>`);
    }
    // Serve static — SPA fallback handled by passing next() for unknown paths
    return serveStatic({ root: distPath })(c, async () => {
      // SPA fallback: unknown path → serve index.html
      return serveStatic({ root: distPath, path: 'index.html' })(c, next);
    });
  };
}
```

import { serveStatic } from '@hono/node-server/serve-static';
import { existsSync } from 'node:fs';
import * as path from 'node:path';
import { fileURLToPath } from 'node:url';
import type { Context, Next } from 'hono';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const distPath = path.resolve(__dirname, '../../packages/dashboard/dist');

export function createStaticHandler() {
  return async (c: Context, next: Next) => {
    if (!existsSync(distPath)) {
      return c.html(`<!DOCTYPE html>
<html lang="en">
<head><meta charset="UTF-8"><title>VFX Familiar</title></head>
<body>
<p>Dashboard not built. Run: <code>npm run build:dashboard</code></p>
</body>
</html>`);
    }
    // Serve static assets; on miss, fall through to index.html for SPA routing
    return serveStatic({ root: distPath })(c, async () => {
      return serveStatic({ root: distPath, path: 'index.html' })(c, next);
    });
  };
}
```
