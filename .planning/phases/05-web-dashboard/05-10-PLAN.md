```tsx
import { h } from 'preact';
import { useState, useEffect } from 'preact/hooks';
import { StatusPill } from '../components/StatusPill.js';
import { JsonBlock } from '../components/JsonBlock.js';
import { EmptyState } from '../components/EmptyState.js';
import { DiffDrawer } from './DiffDrawer.js';
import { fetchProvenance, fetchDiff } from '../lib/api.js';
import type { Version, ProvenanceRecord } from '../types/entities.js';

interface VersionDrawerProps {
  version: Version;
  onClose: () => void;
}

export function VersionDrawer({ version, onClose }: VersionDrawerProps) {
  const [provenance, setProvenance] = useState<ProvenanceRecord[]>([]);
  const [diff, setDiff] = useState<{ before: Version; after: Version } | null>(null);
  const [showDiff, setShowDiff] = useState(false);

  useEffect(() => {
    fetchProvenance(version.id).then(setProvenance).catch(() => setProvenance([]));
  }, [version.id]);

  async function handleViewDiff() {
    if (!diff) {
      const d = await fetchDiff(version.id).catch(() => null);
      setDiff(d);
    }
    setShowDiff(true);
  }

  return (
    <>
      <aside
        class="fixed inset-y-0 right-0 z-10 flex flex-col gap-4 border-l border-[var(--color-border)] bg-[var(--color-bg)] p-4 shadow-xl overflow-y-auto"
        style={{ width: 'var(--drawer-version-width)' }}
        role="dialog"
        aria-label={`Version ${version.label}`}
      >
        <header class="flex items-center justify-between">
          <div class="flex items-center gap-2">
            <h2 class="text-base font-semibold text-[var(--color-text-primary)]">{version.label}</h2>
            <StatusPill status={version.status} />
          </div>
          <div class="flex items-center gap-2">
            <button
              onClick={handleViewDiff}
              class="rounded px-2 py-1 text-xs bg-[var(--color-surface)] text-[var(--color-text-primary)] hover:bg-[var(--color-accent)] hover:text-[var(--color-bg)]"
            >
              View Diff
            </button>
            <button onClick={onClose} aria-label="Close" class="text-[var(--color-text-secondary)]">✕</button>
          </div>
        </header>

        <section>
          <h3 class="mb-2 text-xs uppercase tracking-widest text-[var(--color-text-secondary)]">Provenance</h3>
          {provenance.length === 0 ? (
            <EmptyState message="No provenance records" />
          ) : (
            <ul class="flex flex-col gap-2">
              {provenance.map((record, i) => (
                <li key={i} class="rounded bg-[var(--color-surface)] p-2">
                  <JsonBlock data={record} />
                </li>
              ))}
            </ul>
          )}
        </section>
      </aside>

      {showDiff && (
        <DiffDrawer
          before={diff?.before ?? null}
          after={diff?.after ?? null}
          onClose={() => setShowDiff(false)}
        />
      )}
    </>
  );
}
```

```tsx
import { h } from 'preact';
import { useState, useEffect } from 'preact/hooks';
import { TreeSidebar } from '../components/TreeSidebar.js';
import { VersionCard } from '../components/VersionCard.js';
import { EmptyState } from '../components/EmptyState.js';
import { VersionDrawer } from './VersionDrawer.js';
import { fetchWorkspaces, fetchVersions } from '../lib/api.js';
import { workspaces, selectedShotId } from '../state/hierarchy.js';
import { versions, selectedVersionId } from '../state/versions.js';
import type { Workspace, Version } from '../types/entities.js';

export function HomeView() {
  const [expandedIds, setExpandedIds] = useState<Set<string>>(new Set());

  useEffect(() => {
    fetchWorkspaces()
      .then((ws: Workspace[]) => { workspaces.value = ws; })
      .catch(() => {});
  }, []);

  useEffect(() => {
    const shotId = selectedShotId.value;
    if (!shotId) return;
    fetchVersions(shotId)
      .then((vs: Version[]) => { versions.value = vs; })
      .catch(() => {});
  }, [selectedShotId.value]);

  function toggleExpand(id: string) {
    setExpandedIds((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id); else next.add(id);
      return next;
    });
  }

  const selectedVersion = versions.value.find(
    (v) => v.id === selectedVersionId.value,
  ) ?? null;

  return (
    <div class="flex h-full">
      <TreeSidebar
        workspaces={workspaces.value}
        selectedShotId={selectedShotId.value}
        onSelectShot={(id) => { selectedShotId.value = id; }}
        expandedIds={expandedIds}
        onToggleExpand={toggleExpand}
      />
      <main class="flex flex-1 flex-col gap-2 overflow-y-auto p-4">
        {!selectedShotId.value ? (
          <EmptyState message="Select a shot to view versions" />
        ) : versions.value.length === 0 ? (
          <EmptyState message="No versions yet" />
        ) : (
          <ul class="flex flex-col gap-1">
            {versions.value.map((v) => (
              <li key={v.id}>
                <VersionCard
                  version={v}
                  isSelected={v.id === selectedVersionId.value}
                  onSelect={(id) => { selectedVersionId.value = id; }}
                />
              </li>
            ))}
          </ul>
        )}
      </main>
      {selectedVersion && (
        <VersionDrawer
          version={selectedVersion}
          onClose={() => { selectedVersionId.value = null; }}
        />
      )}
    </div>
  );
}
```
