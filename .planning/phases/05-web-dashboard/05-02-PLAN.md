```typescript
// src/engine/events.ts — produce this file verbatim from RESEARCH.md §5 skeleton
export interface EngineEventMap {
  'version.status_changed': VersionStatusChangedPayload;
  'version.created': VersionCreatedPayload;
  'tag.changed': TagChangedPayload;
  'metadata.changed': MetadataChangedPayload;
  'hierarchy.created': HierarchyCreatedPayload;
}
export class EngineEmitter extends EventEmitter {
  emitEvent<T extends keyof EngineEventMap>(type: T, payload: EngineEventMap[T]): boolean;
  onEvent<T extends keyof EngineEventMap>(type: T, listener: (payload: EngineEventMap[T]) => void): this;
  offEvent<T extends keyof EngineEventMap>(type: T, listener: (payload: EngineEventMap[T]) => void): this;
}
export function createEngineEmitter(): EngineEmitter;

// src/engine/output-downloader.ts — from RESEARCH.md §6
export async function downloadOutput(
  client: ComfyUIClient | null,
  versionId: string,
  outputsDir: string,
  filename: string,
  opts?: { subfolder?: string; type?: string },
): Promise<string | null>;
```

Security note on D-WEBUI-06 MetadataChangedPayload: the `value` field MUST NOT appear in the payload — only `key` is emitted (T-5-02 mitigation).
