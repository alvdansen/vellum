    ```typescript
    import { z } from 'zod';
    import type { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
    import type { Engine } from '../engine/pipeline.js';
    import { TypedError } from '../engine/errors.js';
    import { toolOk, toolError } from './envelope.js';
    import { versionLabel } from '../utils/outputs.js';
    import {
      shapeList,
      MAX_ID_LENGTH,
      MAX_PAGE_SIZE,
      DEFAULT_PAGE_SIZE,
    } from './shape.js';
    import type { Version, Breadcrumb } from '../types/hierarchy.js';
    import type { ProvenanceEvent, DiffResponse } from '../types/provenance.js';

    /**
     * D-PROV-07, D-PROV-08: `version get` — cheap metadata payload. Returns the
     * raw Version row + a derived `version_label` + breadcrumb envelope. Heavy
     * provenance fields (workflow_json / prompt_json) are reached via the
     * explicit `version provenance` opt-in (D-PROV-10).
     */
    const GetInput = z.object({
      action: z.literal('get'),
      version_id: z.string().min(1).max(MAX_ID_LENGTH),
    });

    /**
     * D-PROV-09: `version list` — paginated list of versions for a shot, ordered
     * version_number DESC at the repo layer (Plan 02 `VersionRepo.listByShot`).
     * limit defaults to 20 (Phase 1 D-24), capped at MAX_PAGE_SIZE.
     */
    const ListInput = z.object({
      action: z.literal('list'),
      shot_id: z.string().min(1).max(MAX_ID_LENGTH),
      limit: z.number().int().min(1).max(MAX_PAGE_SIZE).default(DEFAULT_PAGE_SIZE),
      offset: z.number().int().min(0).default(0),
    });

    /**
     * D-PROV-11: `version diff` — two-version comparison. Cross-shot constraint
     * (D-PROV-20) is enforced by the pure diffVersions engine, NOT at this
     * boundary. Not-completed sources (D-PROV-19) surface as VERSION_NOT_COMPLETED.
     */
    const DiffInput = z.object({
      action: z.literal('diff'),
      version_a: z.string().min(1).max(MAX_ID_LENGTH),
      version_b: z.string().min(1).max(MAX_ID_LENGTH),
    });

    /**
     * D-PROV-10: `version provenance` — full chronological event history. Heavy
     * payload (workflow_json + prompt_json + models_json per applicable event).
     */
    const ProvenanceInput = z.object({
      action: z.literal('provenance'),
      version_id: z.string().min(1).max(MAX_ID_LENGTH),
    });

    const VersionInputSchema = z.discriminatedUnion('action', [
      GetInput,
      ListInput,
      DiffInput,
      ProvenanceInput,
    ]);

    /**
     * D-PROV-08: shape a Version entity for `get` responses. Adds `version_label`
     * (mirrors generation-tool's derivation). Does NOT parse outputs_json —
     * `version get` is the cheap-metadata action; heavy payload lives on
     * `version provenance`. outputs_json is still present on the entity when
     * populated — downstream consumers can read it as the raw column — but the
     * tool does not synthesize a typed `outputs` array at this surface.
     */
    function shapeVersionEntity(
      result: { entity: Version; breadcrumb: Breadcrumb },
    ): {
      entity: Version & { version_label: string };
      breadcrumb: Breadcrumb['entries'];
      breadcrumb_text: string;
    } {
      return {
        entity: {
          ...result.entity,
          version_label: versionLabel(result.entity.version_number),
        },
        breadcrumb: result.breadcrumb.entries,
        breadcrumb_text: result.breadcrumb.text,
      };
    }

              return toolError(
                new TypedError('INVALID_INPUT', `Invalid input at 'input.${path}'`),
              );
            }
            return toolError(err);

        },
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Agent (MCP stdio JSON-RPC or HTTP) → Version tool | All input (version_id, shot_id, version_a/version_b, limit/offset) is agent-controlled, untrusted until Zod-validated at the tool boundary. Bounded by MAX_ID_LENGTH and MAX_PAGE_SIZE. |
| Agent (MCP) → Generation tool (reproduce/iterate) | Agent supplies source version_id + overrides + seed + notes. Overrides is a `Record<string, {inputs?, class_type?}>`. Structurally constrained — keys are node ids, no RFC 6902 path traversal surface. |
| Version tool → Engine facade | Tool passes typed validated DTOs; engine enforces existence + state invariants (D-PROV-19 not-completed, D-PROV-20 cross-shot) as defense-in-depth. |
| Generation tool → Engine facade | Tool passes validated iterate overrides; engine owns merging, validation, submit. Tool layer has zero business logic. |
| Engine → ProvenanceRepo | Append-only. Plan 01's repo has no UPDATE/DELETE methods by construction. |
| Tool responses → stdout (stdio transport) | Only MCP JSON-RPC frames. No log lines. Prompt blob values never written to stdout (D-21). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-03-01 | Tampering | `iterate` overrides with unknown node id or `class_type` injection | mitigate | Plan 01's `applyOverrides` rejects unknown node ids with `ITERATE_INVALID_PATCH`; Zod `class_type` is `z.string().optional()` at the tool layer — string-only, no object. Engine-layer `validateWorkflowFormat` post-merge (D-PROV-23) catches any shape escape. No prototype-pollution surface exists because overrides is `Record<string, {...}>`, not a dotted-path string. |
| T-03-03-02 | Tampering | `iterate` overrides with `__proto__` or `constructor` as a node id | mitigate | ComfyUI prompt keys are numeric strings. `applyOverrides` checks `nodeId in out` — `__proto__` is on the prototype chain, not an own property of the prompt blob, so the check fails and `ITERATE_INVALID_PATCH` fires. Zod `z.string().min(1)` does not filter this, but the engine's own-property check does. |
| T-03-03-03 | Information Disclosure | `version provenance` returns full prompt_json to the agent | accept | The prompt blob IS what the agent needs to inspect for provenance. Artists who embed sensitive prompts accept that provenance recall returns the resolved prompt. This is a deliberate feature (PROV-01, PROV-05). Stdio hygiene protects the wire, not the intentional payload. |
| T-03-03-04 | Information Disclosure | Prompt blob logged to stdout during a tool handler | mitigate | The version tool never constructs log lines with prompt_json. The only `console.error` call path is in `generation-tool.ts`'s existing outputs_json-parse-failure branch (stderr — D-21). Stdio-hygiene grep in tool-budget.test.ts / architecture-purity.test.ts catches console.log(prompt_json) drift. |
| T-03-03-05 | Denial of Service | `version list` with extreme offset (e.g., 1e18) | mitigate | Zod `z.number().int().min(0)` permits large integers, but SQLite's `LIMIT/OFFSET` degrades gracefully. Out-of-range numbers produce empty results — no crash. Overflow protection sits at the DB layer. |
| T-03-03-06 | Denial of Service | `iterate` with oversized overrides (10k keys) | mitigate | The overrides record inherits no explicit size cap at the Zod layer beyond `z.string().min(1)` on keys. This is acceptable because: (a) the engine's `applyOverrides` only visits keys that match existing nodes in the source blob, so extra keys surface as `ITERATE_INVALID_PATCH` on the FIRST unknown key (fail-fast), (b) prompt blobs in practice have <50 nodes. Large adversarial payloads fail fast. |
| T-03-03-07 | Elevation of Privilege | `generation iterate` patch that adds a node which executes arbitrary code on ComfyUI Cloud | transfer | ComfyUI Cloud is the execution sandbox for prompts. This MCP server validates structure (`validateWorkflowFormat`), not prompt semantics. Cloud is responsible for worker isolation. |
| T-03-03-08 | Spoofing | Agent calls `reproduce` claiming a version_id that belongs to another tenant | accept | Phase 3 has no multi-tenant auth model (single-user MCP server per CLAUDE.md). Reproduce operates on any local version in the local SQLite DB. When multi-tenant support arrives (v2), asset ownership checks live in the engine. |
| T-03-03-09 | Repudiation | Agent denies it called `reproduce` | mitigate | Every reproduce writes a new version row with `parent_version_id = source`, `lineage_type = 'reproduce'`, plus a `submitted` provenance event. `version provenance` exposes the full event timeline. Chain reconstructable via `versions.parent_version_id`. |
| T-03-03-10 | Information Disclosure | Error messages from `iterate` leak internal blob structure (model paths, seeds) | mitigate | Plan 01's `applyOverrides` error hint includes the valid node id list (`"Valid node ids: 3, 4, 5"`). Node ids alone are not sensitive. Seed values surface only through `version provenance` (explicit opt-in). No error message embeds the full prompt blob. |
| T-03-03-11 | Information Disclosure | Workflow blob included in `version provenance` response reveals CheckpointLoader names, lora names, artist prompts | accept | This is the explicit purpose of the action — provenance includes the authored workflow. D-PROV-10 locks this shape. Stdio hygiene protects the wire; the agent correctly receives the opted-in payload. |
| T-03-03-12 | Tampering | `version diff` with forged version_a/version_b ids that exist on different shots (bypass D-PROV-20) | mitigate | D-PROV-20 is enforced in the pure `diffVersions` engine, NOT at the tool layer. The tool passes both ids through; the engine loads both rows and throws `INVALID_INPUT` if shot_ids differ. Same pattern as Phase 2's `SHOT_NOT_FOUND` check at the engine level — defense in depth. |
