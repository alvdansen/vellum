    ```typescript
    async function handleConfirm(note: string | null) {
      const action = popover.action;
      if (!action) return;
      // Map action to target status (REV-05 → restore=wip; others map 1:1 to status names)
      const toStatus: ShotStatus = action === 'retake' ? 'pending-review'
        : action === 'restore' ? 'wip'
        : action;  // 'approve' | 'hold' | 'omit' map directly
      // Strict mapping per REQUIREMENTS + UI-SPEC: approve→'approved', hold→'on-hold'.
      // Refine the mapping table:
      //   approve → 'approved'
      //   retake  → 'pending-review'
      //   hold    → 'on-hold'
      //   omit    → 'omit'
      //   restore → 'wip'
      actionInFlight.value = action;
      try {
        await setShotStatus(shotId, { to_status: toStatus, note, changed_by: 'user' });
        setPopover({ open: false, action: null });
        // SSE will arrive shortly and update shotGrid; panel header refreshes
        setError(null);
      } catch (err: any) {
        setError({ message: `${REVIEW_PANEL_ACTION_FAIL_PREFIX}${err?.message ?? 'server error'} — retry` });
        // Popover stays open so user can edit note and retry
      } finally {
        actionInFlight.value = null;
      }
    }
    ```

    ```typescript
        Promise.all([

            statusHistory: historyResp.history,
          });

          if (!alive) return;
          setLoadError(REVIEW_HISTORY_FETCH_ERROR);
        });
        return () => { alive = false; };
      }, [shotId]);

      if (data === null) {
    ```
