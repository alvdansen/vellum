    ```typescript
    export function ABCompareHost() {
      const isOpen = compareModalOpen.value;
      const selection = compareSelection.value;
      const shotId = activeReviewShotId.value;

      // Reset compareSelection when shotId changes (Pitfall 6)
      useEffect(() => {
        compareSelection.value = { a: null, b: null };
      }, [shotId]);

      const [versions, setVersions] = useState<{ a?: any; b?: any } | null>(null);

      // Fetch the two versions' metadata for v-number labels.
      useEffect(() => {
        if (!isOpen || selection.a === null || selection.b === null || shotId === null) {
          setVersions(null);
          return;
        }
        let alive = true;
        fetchVersions(shotId, {}).then((resp: any) => {
          if (!alive) return;
          const list: any[] = resp.versions ?? resp.items ?? [];
          const a = list.find(v => v.id === selection.a);
          const b = list.find(v => v.id === selection.b);
          if (a && b) setVersions({ a, b });
        });
        return () => { alive = false; };
      }, [isOpen, selection.a, selection.b, shotId]);

      if (!isOpen) return null;
      if (selection.a === null || selection.b === null) return null;
      if (versions === null) {
        // Fetching version metadata — render a minimal loading backdrop
        return (
          <div class="fixed inset-0 z-30 flex items-center justify-center bg-black/60" aria-busy="true" />
        );
      }

      const shotName = shotGrid.value?.shots.find(s => s.id === shotId)?.name ?? shotId;

      return (
        <ABCompareView
          shotName={shotName}
          versionA={versions.a}
          versionB={versions.b}
          onClose={() => { compareModalOpen.value = false; }}
        />
      );
    }
    ```
