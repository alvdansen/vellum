    ```typescript
    /**
     * Phase 15 (D-CTX-3 / D-CTX-4 / D-CTX-6) — derive the ingredients tuple
     * for a version. Reads:
     *   - The version row (parent_version_id + lineage_type)
     *   - The latest 'completed' provenance event (resolved prompt blob + seed)
     *   - The parent's latest manifest_signed event (manifest_sha256 lookup)
     *   - The on-disk component image bytes via hashComponentBytes
     *
     * Returns null when there is no completed event yet (no prompt blob to
     * walk) — the manifest still gets built with no ingredients (Phase 14
     * single-c2pa.created shape). All other failure modes degrade
     * gracefully (parent_manifest_pending / component_unavailable).
     *
     * Hot-path safety: the parent-hash lookup is one indexed scan
     * (getLatestManifestSignedEvent already LIMITs to 50 rows). Component
     * hashing is bounded by the number of IMAGE_INPUT_CLASS_TYPES nodes in
     * the prompt blob — typically &lt;= 5 in production workflows.
     */
    private async buildIngredientsForVersion(
      versionId: string,
    ): Promise&lt;{
      parentOf: ParentIngredient | null;
      componentOf: ComponentIngredient[];
      componentHashes: Map&lt;string, ComponentHashOutcome&gt;;
      inputTo: InputAssertion;
    } | null&gt;
    {
      const version = this.versionRepo.getVersion(versionId);
      if (!version) return null;
      const completed = this.provenanceRepo.getLatestCompletedEvent(versionId);
      if (!completed?.prompt_json) return null;
      let promptBlob: Record&lt;string, unknown&gt;;
      try {
        const parsed = JSON.parse(completed.prompt_json);
        if (!parsed || typeof parsed !== 'object' || Array.isArray(parsed)) return null;
        promptBlob = parsed as Record&lt;string, unknown&gt;;
      } catch {
        return null;
      }

      // parentOf — read parent's latest manifest_signed event, prefer the
      // manifest_sha256 field. The accessor takes (versionId, filename) but
      // for the parent-manifest-hash lookup we need ANY filename — the
      // parent typically has a single output. Read the parent version's
      // outputs_json to extract its first filename, then call the accessor.
      const parentOf = extractParentIngredient(version, (parentId) =&gt; {
        const parent = this.versionRepo.getVersion(parentId);
        if (!parent?.outputs_json) return null;
        let parentFilename: string | null = null;
        try {
          const parsed = JSON.parse(parent.outputs_json) as Array&lt;{ filename?: string }&gt;;
          parentFilename = Array.isArray(parsed) ? parsed[0]?.filename ?? null : null;
        } catch {
          return null;
        }
        if (!parentFilename) return null;
        const event = this.provenanceRepo.getLatestManifestSignedEvent(parentId, parentFilename);
        if (!event || event.signed !== true || !event.manifest_sha256) return null;
        return event.manifest_sha256;
      });

      // componentOf — pure walk + impure hash per ingredient.
      const componentOf = extractComponentIngredients(promptBlob);
      const componentHashes = new Map&lt;string, ComponentHashOutcome&gt;();
      for (const c of componentOf) {
        const outcome = await hashComponentBytes(this.outputRoot, versionId, c.input_filename);
        componentHashes.set(c.node_id, outcome);
      }

      // inputTo — extract structured payload + seed from the completed event.
      const inputTo = extractInputAssertion(promptBlob, completed.seed ?? null);

      return { parentOf, componentOf, componentHashes, inputTo };
    }
    ```

    ```typescript
    /**
     * Phase 15 — stream-hash a file's SHA-256 without loading bytes into
     * memory. Used by Engine.signOutput's embed-file path to hash the signed
     * bytes that were written to a temp file. Returns null on any I/O error
     * — the caller (manifest_signed event) records null cleanly.
     */
    async function streamSha256(filePath: string): Promise&lt;string | null&gt; {
      try {
        return await new Promise&lt;string&gt;((resolve, reject) =&gt; {
          const hash = createHash('sha256');
          const stream = createReadStream(filePath);
          stream.on('error', reject);
          stream.on('data', (chunk) =&gt; hash.update(chunk));
          stream.on('end', () =&gt; resolve(hash.digest('hex')));
        });
      } catch {
        return null;
      }
    }

     * NULL ingredients (no completed event yet) → all-zero summary so audits

      }
      return {
    ```
