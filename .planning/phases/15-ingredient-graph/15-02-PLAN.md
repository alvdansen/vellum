      const description = describePrimaryModel(opts.primaryModel);
      const assertions: ManifestAssertion[] = [
        {
          label: 'c2pa.actions',
          data: {
            actions: [
              {
                action: 'c2pa.created',
                digitalSourceType: IPTC_TRAINED_ALGORITHMIC_MEDIA,
                softwareAgent: { name: 'ComfyUI', version: opts.comfyuiVersion },
                parameters: { description },
              },
            ],
          },
        },
      ];

      // Phase 15 — layer ingredient assertions when provided. Backward-compat
      // invariant: omitted ingredients leaves assertions[] at exactly the
      // Phase 14 single-c2pa.actions shape.
      if (opts.ingredients) {
        const { parentOf, componentOf, componentHashes, inputTo } = opts.ingredients;

        // parentOf — single assertion when present.
        if (parentOf !== null) {
          assertions.push(buildParentAssertion(parentOf));
        }

        // componentOf — one assertion per ingredient. Determinism: input
        // array is already sorted by node_id (extractComponentIngredients
        // contract) so we preserve that order.

          assertions.push(buildComponentAssertion(c, componentHashes));
        }

        // inputTo — single vendor-namespaced assertion.
        assertions.push({ label: 'vfx_familiar.input', data: inputTo });
      }

      return {
        claim_generator: `vfx-familiar/${opts.appVersion} c2pa-node/${C2PA_NODE_VERSION}`,
        format: opts.mimeType,
        title: `Version ${opts.versionId}`,
        assertions,

    Add the two private helpers (place them BELOW the public buildManifestDefinition function, ABOVE describePrimaryModel for organisation):

    ```typescript
    /**
     * Phase 15 — build the parentOf c2pa.ingredient assertion. When
     * parent_unavailable is set, hash is null and metadata records the
     * dangling-reference reason (D-CTX-6 + criterion #5).
     */
    function buildParentAssertion(p: ParentIngredient): IngredientAssertion {
      const metadata: Record&lt;string, string | number | null&gt; = {
        version_id: p.parent_version_id,
        lineage_type: p.lineage_type,
      };
      if (p.parent_unavailable !== null) {
        metadata.parent_unavailable = p.parent_unavailable;
      }
      return {
        label: 'c2pa.ingredient',
        data: {
          title: `Parent ${p.parent_version_id}`,
          relationship: 'parentOf',
          hash: p.manifest_hash !== null ? `sha256:${p.manifest_hash}` : null,
          metadata,
        },
      };
    }

    /**
     * Phase 15 — build a componentOf c2pa.ingredient assertion. T-15-04
     * mitigation: metadata.input_filename carries the BASENAME only.
     * componentHashes is an out-of-band lookup keyed by ComponentIngredient.node_id;
     * when missing OR when the entry carries component_unavailable, the
     * assertion records the dangling-reference state.
     */
    function buildComponentAssertion(
      c: ComponentIngredient,
      componentHashes: ReadonlyMap&lt;string, ComponentHashOutcome&gt; | undefined,
    ): IngredientAssertion {
      const outcome = componentHashes?.get(c.node_id);
      const metadata: Record&lt;string, string | number | null&gt; = {
        node_id: c.node_id,
        role: c.role,
        // T-15-04: basename only; never an absolute path. The extractor
        // already returns basenames (Comfy stores basenames in the prompt
        // blob's image field), but defence-in-depth — strip any directory
        // components defensively here too.
        input_filename: stripToBasename(c.input_filename),
      };
      let hash: string | null = null;
      if (outcome &amp;&amp; 'hash' in outcome) {
        hash = `sha256:${outcome.hash}`;
      } else if (outcome &amp;&amp; 'component_unavailable' in outcome) {
        metadata.component_unavailable = outcome.component_unavailable;
      } else {
        // No outcome supplied — caller never attempted hashing for this
        // ingredient. Record as 'pending' rather than silently dropping.
        metadata.component_unavailable = 'hash_pending';
      }
      return {
        label: 'c2pa.ingredient',
        data: {
          title: `${c.role} image (${stripToBasename(c.input_filename)})`,
          relationship: 'componentOf',
          hash,
          metadata,
        },
      };
    }

    /** T-15-04 defence-in-depth: trust no caller, strip to basename. Pure
     *  string ops — no path module dependency to keep this module pure-string. */
    function stripToBasename(p: string): string {
      // Take the segment after the LAST '/' or '\\' separator. Empty input
      // returns empty (caller already filtered nulls).
      const lastSlash = p.lastIndexOf('/');
      const lastBackslash = p.lastIndexOf('\\');
      const idx = Math.max(lastSlash, lastBackslash);
      if (idx === -1) return p;
      return p.slice(idx + 1);
    }
    ```
