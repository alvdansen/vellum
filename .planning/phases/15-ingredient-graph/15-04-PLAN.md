    it('Test 1 (criterion #4): v1 → v2 (reproduce + control image) → v3 (iterate) ingredient graph traces back through every step', async () =&gt; {
      // --- Arrange: hierarchy + workflow blob with LoadImage + LoRA ---
      const ws = engine.createWorkspace('test_ws');
      const proj = engine.createProject(ws.entity.id, 'test_proj');
      const seq = engine.createSequence(proj.entity.id, 'test_seq');
      const shot = engine.createShot(seq.entity.id, 'sh010');

      const v1Workflow = makeWorkflow({ /* CheckpointLoaderSimple + KSampler + CLIPTextEncode + LoraLoader */ });

      // --- Act: v1 generation ---
      const v1Res = await engine.submitGeneration(shot.entity.id, v1Workflow);
      // Drive completion via FakeComfyUIClient + getGenerationStatus polling.
      await drivePolling(v1Res.entity.id);
      // Sign v1's output.
      const v1Bytes = await engine.signOutput(v1Res.entity.id, 'out.png', { bytes: v1FixturePng });
      expect(v1Bytes.signed).not.toBeNull();
      const v1ManifestEvent = engine.getC2paStatusForVersion(v1Res.entity.id, 'out.png');
      expect(v1ManifestEvent!.manifest_sha256).toBeDefined();
      const v1Sha = v1ManifestEvent!.manifest_sha256!;

      // --- Act: v2 reproduction (adds control image + LoRA) ---
      // Note: reproduceVersion in production reuses the SAME workflow as v1.
      // For this test we want v2 to ADD a LoadImage component, so we drive
      // reproduce + then patch the prompt blob via the FakeComfyUIClient's
      // canned 'completed' response. (If reproduceVersion enforces verbatim
      // workflow replay, use iterateFromVersion with overrides instead and
      // adjust the test's lineage_type assertion accordingly.)
      const v2Res = await engine.reproduceVersion(v1Res.entity.id);
      await drivePolling(v2Res.entity.id);

      // Pre-write the control image to outputRoot so hashComponentBytes finds it.
      const controlBytes = randomBytes(1024);
      const controlSha = createHash('sha256').update(controlBytes).digest('hex');
      await mkdir(join(outputRoot, v2Res.entity.id), { recursive: true });
      await writeFile(join(outputRoot, v2Res.entity.id, 'control.png'), controlBytes);

      // Sign v2.
      const v2Bytes = await engine.signOutput(v2Res.entity.id, 'out.png', { bytes: v2FixturePng });
      expect(v2Bytes.signed).not.toBeNull();
      const v2ManifestEvent = engine.getC2paStatusForVersion(v2Res.entity.id, 'out.png');
      const v2Sha = v2ManifestEvent!.manifest_sha256!;

      // --- Act: v3 iterate from v2 ---
      const v3Res = await engine.iterateFromVersion(v2Res.entity.id, undefined, 999);
      await drivePolling(v3Res.entity.id);
      const v3Bytes = await engine.signOutput(v3Res.entity.id, 'out.png', { bytes: v3FixturePng });
      expect(v3Bytes.signed).not.toBeNull();

      // --- Assert: read v3's manifest, walk ingredients ---
      const { createC2pa } = await import('c2pa-node');
      const c2pa = createC2pa();
      const v3Store = await c2pa.read({ buffer: v3Bytes.signed!, mimeType: 'image/png' });
      expect(v3Store!.active_manifest).not.toBeNull();
      const v3Assertions = (v3Store!.active_manifest!.assertions ?? []) as Array&lt;{label?:string; data?:any}&gt;;

      // v3 must carry parentOf → v2.
      const v3Parent = v3Assertions.find(a =&gt; a.label === 'c2pa.ingredient' &amp;&amp; a.data?.relationship === 'parentOf');
      expect(v3Parent).toBeDefined();
      expect(v3Parent!.data!.hash).toBe(`sha256:${v2Sha}`);
      expect(v3Parent!.data!.metadata.lineage_type).toBe('iterate');

      // --- Assert: read v2's manifest, walk ingredients, prove componentOf for control image + parentOf for v1 ---
      const v2Store = await c2pa.read({ buffer: v2Bytes.signed!, mimeType: 'image/png' });
      const v2Assertions = (v2Store!.active_manifest!.assertions ?? []) as Array&lt;{label?:string; data?:any}&gt;;

      // v2 → v1 parentOf.
      const v2Parent = v2Assertions.find(a =&gt; a.label === 'c2pa.ingredient' &amp;&amp; a.data?.relationship === 'parentOf');
      expect(v2Parent).toBeDefined();
      expect(v2Parent!.data!.hash).toBe(`sha256:${v1Sha}`);
      expect(v2Parent!.data!.metadata.lineage_type).toBe('reproduce');

      // v2 componentOf for control.png.
      const v2Component = v2Assertions.find(a =&gt; a.label === 'c2pa.ingredient' &amp;&amp; a.data?.relationship === 'componentOf');
      expect(v2Component).toBeDefined();
      expect(v2Component!.data!.hash).toBe(`sha256:${controlSha}`);
      expect(v2Component!.data!.metadata.input_filename).toBe('control.png');

      // --- Assert: v1 has NO parentOf (top of chain). LoRA fingerprint flows in via Phase 13's primary model assertion. ---
      const v1Store = await c2pa.read({ buffer: v1Bytes.signed!, mimeType: 'image/png' });
      const v1Assertions = (v1Store!.active_manifest!.assertions ?? []) as Array&lt;{label?:string; data?:any}&gt;;
      expect(v1Assertions.find(a =&gt; a.label === 'c2pa.ingredient' &amp;&amp; a.data?.relationship === 'parentOf')).toBeUndefined();
    });
