    ```typescript
    /**
     * Phase 14 — D-CTX-3 native-embed format routing.
     *
     * Pure function — no I/O, no side effects. Returns one of three routes:
     *  - { mode: 'embed', mimeType }   — PNG/JPEG/MP4/WebP via c2pa-node native embed
     *  - { mode: 'sidecar', mimeType } — EXR/PSD/TIFF via sidecar .c2pa file alongside
     *  - { mode: 'unsupported' }       — unknown extension; engine returns original bytes
     *
     * Extension matching is case-insensitive (T-14-08 mitigation: ComfyUI Cloud
     * sometimes returns mixed-case filenames). The mimeType strings are the C2PA
     * standard registrations and match what c2pa-node expects in its asset.mimeType
     * field.
     */
    export type FormatRoute =
      | { mode: 'embed'; mimeType: string }
      | { mode: 'sidecar'; mimeType: string }
      | { mode: 'unsupported' };

    /** D-CTX-3 native-embed list. c2pa-node v0.5.x supports buffer signing for
     *  JPEG + PNG only; MP4 + WebP go through the file-path signing API. The
     *  routing decision is the same — mode='embed' — regardless of the underlying
     *  c2pa-node call shape. The signer wrapper handles the buffer-vs-file split. */
    const EMBED_TABLE: Record<string, string> = {
      '.png': 'image/png',
      '.jpg': 'image/jpeg',
      '.jpeg': 'image/jpeg',
      '.mp4': 'video/mp4',
      '.webp': 'image/webp',
    };

    const SIDECAR_TABLE: Record<string, string> = {
      '.exr': 'image/x-exr',
      '.psd': 'image/vnd.adobe.photoshop',
      '.tif': 'image/tiff',
      '.tiff': 'image/tiff',
    };

    export const EMBED_FORMATS: readonly string[] = Object.values(EMBED_TABLE);
    export const SIDECAR_FORMATS: readonly string[] = Object.values(SIDECAR_TABLE);

    export function routeFormat(filename: string): FormatRoute {
      const dot = filename.lastIndexOf('.');
      if (dot < 0) return { mode: 'unsupported' };
      const ext = filename.slice(dot).toLowerCase();
      if (ext in EMBED_TABLE) return { mode: 'embed', mimeType: EMBED_TABLE[ext]! };
      if (ext in SIDECAR_TABLE) return { mode: 'sidecar', mimeType: SIDECAR_TABLE[ext]! };
      return { mode: 'unsupported' };
    }
    ```

    ```typescript
    ): Promise<LoadedSigner> {
      let certPemBytes: Buffer;
      let privateKeyPemBytes: Buffer;
      try {
        [certPemBytes, privateKeyPemBytes] = await Promise.all([
          readFile(certPemPath),
          readFile(privateKeyPemPath),
        ]);
      } catch (err) {
        throw new TypedError(
          'C2PA_SIGNER_LOAD_FAILED',
          `Failed to read cert or key PEM: ${(err as Error).message}`,
          'Verify VFX_FAMILIAR_C2PA_CERT_PEM_PATH and VFX_FAMILIAR_C2PA_PRIVATE_KEY_PEM_PATH point to readable files.',
        );
      }

      let certSubjectSummary: string;
      try {
        const cert = new X509Certificate(certPemBytes);
        // Use the subject's CN or O — falling back to a fingerprint prefix.
        // Subject is a comma-separated DN string like 'CN=vfx-familiar dev,O=example'.
        const subject = cert.subject ?? '';
        const cn = subject.match(/CN=([^,]+)/)?.[1];
        const o = subject.match(/O=([^,]+)/)?.[1];
        certSubjectSummary = cn ?? o ?? `fp:${cert.fingerprint256.slice(0, 16)}`;
      } catch (err) {
        throw new TypedError(
          'C2PA_SIGNER_LOAD_FAILED',
          `Cert PEM is not parseable as X.509: ${(err as Error).message}`,
          'Regenerate the cert via scripts/gen-dev-c2pa-cert.mts (dev) or contact your CA (prod).',
        );
      }

      let c2pa: C2pa;
      try {
        c2pa = createC2pa({
          signer: {
            type: 'local',
            certificate: certPemBytes,
            privateKey: privateKeyPemBytes,
            algorithm: SigningAlgorithm.ES256,
          },
        });
      } catch (err) {
        throw new TypedError(
          'C2PA_SIGNER_LOAD_FAILED',
          `c2pa-node createC2pa rejected the cert/key pair: ${(err as Error).message}`,
          'Verify cert + key match (ES256 / P-256 expected) and the cert PEM contains the full chain if needed.',
        );
      }

      return { c2pa, certSubjectSummary };
    }

    /** Sign + embed a manifest into a buffer asset (JPEG / PNG only). */
    export async function signEmbed(
      buffer: Buffer,
      mimeType: string,
      manifestDef: ManifestDefinition,
      signer: LoadedSigner,
    ): Promise<Buffer> {
      if (!BUFFER_API_MIMETYPES.has(mimeType)) {
        throw new TypedError(
          'C2PA_SIGNING_FAILED',
          `Buffer-API signing not supported for ${mimeType} — use signEmbedFromPath instead`,
          'c2pa-node v0.5.x supports buffer signing only for image/jpeg and image/png. MP4 and WebP must use the file-path API.',
        );
      }
      try {
        const result = await signer.c2pa.sign({
          asset: { buffer, mimeType },
          manifest: new ManifestBuilder(manifestDef as never),
        });
        return result.signedAsset.buffer;
      } catch (err) {
        throw new TypedError(
          'C2PA_SIGNING_FAILED',
          `c2pa-node sign() rejected the asset: ${(err as Error).message}`,
          'Caller should degrade gracefully (return original bytes) per D-CTX-9.',
        );
      }
    }

    /** Sign + embed a manifest into a file asset (MP4, WebP, JPEG, PNG — file form).
     *  Required for MP4 + WebP per c2pa-node v0.5.x buffer-API limits. */
    export async function signEmbedFromPath(
      srcPath: string,
      destPath: string,
      mimeType: string,
      manifestDef: ManifestDefinition,
      signer: LoadedSigner,
    ): Promise<void> {
      try {
        await signer.c2pa.sign({
          asset: { path: srcPath, mimeType },
          manifest: new ManifestBuilder(manifestDef as never),
          thumbnail: false,
          options: { outputPath: destPath },
        });
      } catch (err) {
        throw new TypedError(
          'C2PA_SIGNING_FAILED',
          `c2pa-node sign() (file API) rejected the asset: ${(err as Error).message}`,
          'Caller should degrade gracefully (return original bytes / skip sidecar) per D-CTX-9.',
        );
      }
    }
    ```
