    import { accessSync, statSync, constants as fsConstants } from 'node:fs';

    /**
     * Phase 14 — PROV-V-01 (D-CTX-2). Read cert + key paths from env, validate
     * existence + readability + non-empty, and return a C2paConfig OR null
     * (signing disabled). Throws TypedError('C2PA_CONFIG_INVALID', ...) when
     * exactly one var is set, or either path is unreadable / empty. Throwing
     * BEFORE any transport connect or tool register is the parity-with-Phase-10
     * MIGRATION_PENDING-typed-error pattern (D-CTX-9 graceful-fail applies at
     * download time; boot-time misconfig fails loud).

      const certPath = process.env.VFX_FAMILIAR_C2PA_CERT_PEM_PATH;
      const keyPath = process.env.VFX_FAMILIAR_C2PA_PRIVATE_KEY_PEM_PATH;

      // Both unset → signing disabled (D-CTX-2 graceful degradation).
      if (!certPath && !keyPath) return null;

      // Exactly one set → misconfig.
      if (!certPath || !keyPath) {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          'Both VFX_FAMILIAR_C2PA_CERT_PEM_PATH and VFX_FAMILIAR_C2PA_PRIVATE_KEY_PEM_PATH must be set together, or both unset',
          'Set both paths to enable C2PA signing, or unset both to run with signing disabled.',
        );
      }

      // Validate cert path.
      try {
        accessSync(certPath, fsConstants.R_OK);
      } catch {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA cert PEM not readable at ${certPath}`,
          'Ensure the file exists and the server process has read permission. Run scripts/gen-dev-c2pa-cert.mts for a local dev cert.',
        );
      }
      const certStat = statSync(certPath);
      if (certStat.size === 0) {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA cert PEM is empty at ${certPath}`,
          'The cert file exists but is zero bytes. Regenerate or provide a real PEM.',
        );
      }

      // Validate key path (same pattern). NEVER read the key bytes here — only
      // stat. T-14-01 mitigation: key bytes never touch a logger or env-dump.
      try {
        accessSync(keyPath, fsConstants.R_OK);
      } catch {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA private key PEM not readable at ${keyPath}`,
          'Ensure the file exists and the server process has read permission.',
        );
      }
      const keyStat = statSync(keyPath);
      if (keyStat.size === 0) {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA private key PEM is empty at ${keyPath}`,
          'The key file exists but is zero bytes. Regenerate or provide a real PEM.',
        );
      }

      // T-14-04 mitigation: warn if key file is world-readable (mode bits
      // beyond owner). Does NOT throw — many test fixtures land at 0644.
      const mode = keyStat.mode &amp; 0o777;
      if (mode &amp; 0o077) {
        console.error(
          `vfx-familiar: WARNING — C2PA private key file at ${keyPath} has permissive mode 0${mode.toString(8)}; tighten to 0600 for production.`,
        );
      }

      return { certPemPath: certPath, privateKeyPemPath: keyPath };
    }

    /**
     * Phase 14 — PROV-V-01. Local dev C2PA cert generator.
     *
     * DEV-ONLY. Generates a self-signed ES256 cert + private key pair to

    const { publicKey, privateKey } = generateKeyPairSync('ec', {

    // Self-signed cert via node:crypto's X.509 builder (Node 20+).
    // ... uses x509.X509Certificate or a minimal manual DER assembly,
