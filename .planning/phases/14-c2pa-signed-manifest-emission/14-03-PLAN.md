Sidecar emission strategy: c2pa-node v0.5.x does NOT have a built-in `signSidecar`. For
EXR/PSD/TIFF (formats c2pa-node cannot embed into), the sidecar bytes come from the
`signedManifest` field of a sign() call against a SUPPORTED placeholder format
(typically a tiny 1x1 PNG fixture wrapped around the same manifest definition). The
output's actual bytes stay unchanged on disk; the sidecar carries the manifest only.

    // ================================================================
    // PHASE 14 — C2PA SIGNED MANIFEST EMISSION (PROV-V-01 / -02 / -05)
    // ================================================================

    private signerCache: { signer: LoadedSigner } | null = null;
    private signerLoadFailedReason: string | null = null;

    /**
     * Phase 14 — PROV-V-01. Sign one output for a version. Always fires a
     * manifest_signed provenance event (success or failure). Returns
     * { signed: Buffer; sidecar: Buffer | null } describing what should land
     * on disk. The CALLER (output-downloader) writes the bytes; this method
     * is pure-ish — it only reads/writes the DB (provenance event), never
     * touches the output filesystem.
     *
     * Failure modes (D-CTX-9 graceful):
     *   - signing disabled (no c2paConfig)        → returns input verbatim + manifest_signed:false
     *   - unsupported format                      → returns input verbatim + manifest_signed:false
     *   - cert load failed                        → returns input verbatim + manifest_signed:false
     *   - sign call threw                         → returns input verbatim + manifest_signed:false
     *   - file too large for buffer API           → falls through to sidecar mode
     *
     * T-14-06 mitigation: every c2pa-node call is in the signer wrapper which
     * already wraps in try/catch. Any TypedError that escapes signEmbed /
     * signEmbedFromPath is caught here and converted to a graceful-fail event.
     */
    async signOutput(
      versionId: string,
      filename: string,
      bytes: Buffer,
    ): Promise<{ signed: Buffer; sidecar: Buffer | null }> {
      const signedAt = new Date().toISOString();

      // Path 1: signing disabled (no c2paConfig).
      if (this.c2paConfig === null) {
        this.provenanceRepo.appendManifestSignedEvent(versionId, {
          format: '',
          sidecar: false,
          signed: false,
          cert_subject_summary: '',
          signed_at: signedAt,
          status_reason: 'signing_disabled',
        });
        return { signed: bytes, sidecar: null };
      }

      // Path 2: format routing.
      const route = routeFormat(filename);
      if (route.mode === 'unsupported') {
        this.provenanceRepo.appendManifestSignedEvent(versionId, {
          format: '',
          sidecar: false,
          signed: false,
          cert_subject_summary: '',
          signed_at: signedAt,
          status_reason: 'unsupported_format',
        });
        return { signed: bytes, sidecar: null };
      }

      // Path 3: lazy signer load (T-14-06 — corrupted cert surfaces here).
      const signer = await this.getOrLoadSigner();
      if (signer === null) {
        this.provenanceRepo.appendManifestSignedEvent(versionId, {
          format: route.mimeType,
          sidecar: false,
          signed: false,
          cert_subject_summary: '',
          signed_at: signedAt,
          status_reason: 'cert_load_failed',
        });
        return { signed: bytes, sidecar: null };
      }

      // Path 4: build manifest definition.
      const fingerprints = this.provenanceRepo.getLatestFingerprints(versionId);
      const primaryModel = derivePrimaryModel(fingerprints);
      const appVersion = await this.readAppVersion();
      const manifestDef = buildManifestDefinition({
        versionId,
        mimeType: route.mimeType,
        primaryModel,
        comfyuiVersion: null, // Phase 14 deferred: ComfyUI server version capture is a follow-up
        appVersion,
      });

      // Path 5: route to embed (buffer or file) or sidecar.
      try {
        if (route.mode === 'embed') {
          // Size cap (T-14-07).
          if (bytes.length > BUFFER_SIGNING_MAX_BYTES &amp;&amp; route.mimeType !== 'video/mp4') {
            // Fall through to sidecar mode for very large files (rare for non-MP4).
            const sidecarBytes = await this.deriveSidecarManifestBytes(manifestDef, signer);
            this.provenanceRepo.appendManifestSignedEvent(versionId, {
              format: route.mimeType,
              sidecar: true,
              signed: true,
              cert_subject_summary: signer.certSubjectSummary,
              signed_at: signedAt,
              status_reason: '',
            });
            return { signed: bytes, sidecar: sidecarBytes };
          }
          // Buffer API for JPEG + PNG; file API for MP4 + WebP (per c2pa-node v0.5.x).
          if (route.mimeType === 'image/jpeg' || route.mimeType === 'image/png') {
            const signedBytes = await signEmbed(bytes, route.mimeType, manifestDef, signer);
            this.provenanceRepo.appendManifestSignedEvent(versionId, {
              format: route.mimeType,
              sidecar: false,
              signed: true,
              cert_subject_summary: signer.certSubjectSummary,
              signed_at: signedAt,
              status_reason: '',
            });
            return { signed: signedBytes, sidecar: null };
          }
          // File API: write input to a temp file, sign to a second temp file, read back.
          const signedBytes = await this.signViaTempFiles(
            bytes,
            route.mimeType,
            manifestDef,
            signer,
          );
          this.provenanceRepo.appendManifestSignedEvent(versionId, {
            format: route.mimeType,
            sidecar: false,
            signed: true,
            cert_subject_summary: signer.certSubjectSummary,
            signed_at: signedAt,
            status_reason: '',
          });
          return { signed: signedBytes, sidecar: null };
        }
        // mode === 'sidecar' — c2pa-node cannot embed into EXR/PSD/TIFF.
        // Derive sidecar manifest bytes from a placeholder PNG sign call.
        const sidecarBytes = await this.deriveSidecarManifestBytes(manifestDef, signer);
        this.provenanceRepo.appendManifestSignedEvent(versionId, {
          format: route.mimeType,
          sidecar: true,
          signed: true,
          cert_subject_summary: signer.certSubjectSummary,
          signed_at: signedAt,
          status_reason: '',
        });
        return { signed: bytes, sidecar: sidecarBytes };
      } catch (err) {
        // T-14-06 + D-CTX-9: any TypedError from the c2pa-node wrappers lands here.
        console.error(
          `vfx-familiar: C2PA signing failed for ${versionId}/${filename}: ${(err as Error).message}`,
        );
        this.provenanceRepo.appendManifestSignedEvent(versionId, {
          format: route.mimeType,
          sidecar: false,
          signed: false,
          cert_subject_summary: signer.certSubjectSummary,
          signed_at: signedAt,
          status_reason: 'sign_call_failed',
        });
        return { signed: bytes, sidecar: null };
      }
    }

    /** Lazy-load signer once per process. Returns null if loadSigner throws. */
    private async getOrLoadSigner(): Promise<LoadedSigner | null> {
      if (this.signerCache !== null) return this.signerCache.signer;
      if (this.signerLoadFailedReason !== null) return null;
      if (this.c2paConfig === null) return null;
      try {
        const signer = await loadSigner(
          this.c2paConfig.certPemPath,
          this.c2paConfig.privateKeyPemPath,
        );
        this.signerCache = { signer };
        return signer;
      } catch (err) {
