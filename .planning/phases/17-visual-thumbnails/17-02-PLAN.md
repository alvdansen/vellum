`realSpawnFfmpeg` (the production implementation of the test-injectable seam):
```typescript
function realSpawnFfmpeg(bin: string, args: string[], timeoutMs: number): Promise<void> {
  return new Promise((resolve, reject) => {
    const proc = spawn(bin, args, { stdio: ['ignore', 'ignore', 'pipe'] });
    let stderr = '';
    proc.stderr.on('data', (d) => { stderr += d.toString(); if (stderr.length > 8192) stderr = stderr.slice(-4096); });
    const timer = setTimeout(() => { try { proc.kill('SIGKILL'); } catch {} reject(new Error('ffmpeg_timeout')); }, timeoutMs);
    proc.once('error', (err) => { clearTimeout(timer); reject(err); });
    proc.once('close', (code) => { clearTimeout(timer); if (code === 0) resolve(); else reject(new Error(`ffmpeg_failed:${code}:${stderr.slice(-512)}`)); });
  });
}
```
