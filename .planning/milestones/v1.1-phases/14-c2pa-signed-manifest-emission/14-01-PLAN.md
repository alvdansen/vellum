    ```typescript
    import { accessSync, statSync, realpathSync, constants as fsConstants } from 'node:fs';
    import { basename, sep } from 'node:path';
    import { TypedError } from '../engine/errors.js';
    import type { C2paConfig } from '../types/c2pa.js';

    export const DEFAULT_C2PA_CERT_ROOT_HINT =
      'Set VFX_FAMILIAR_C2PA_CERT_ROOT to override the allowlist root (defaults to cwd).';

    /**
     * Phase 14 — PROV-V-01 (D-CTX-2). Concern #4 mitigation: realpath + allowlist
     * containment. Reads cert + key paths from env, resolves symlinks via realpathSync,
     * asserts the resolved paths live inside the allowlist root (cwd by default,
     * override via VFX_FAMILIAR_C2PA_CERT_ROOT), validates existence + readability +
     * non-empty, and returns a C2paConfig OR null (signing disabled). Throws
     * TypedError('C2PA_CONFIG_INVALID', ...) when validation fails.
     *
     * Throwing BEFORE any transport connect or tool register is the parity-with-
     * Phase-10 MIGRATION_PENDING-typed-error pattern (D-CTX-9 graceful-fail applies
     * at download time; boot-time misconfig fails loud).
     *
     * **Path leak hygiene:** Error messages and the boot success log emit ONLY the
     * basename of the cert/key files, never the full path. The full resolved paths
     * are stored in the returned C2paConfig for use by Plan 14-02's signer module.
     */
    export function loadC2paConfigFromEnv(): C2paConfig | null {
      const certPathRaw = process.env.VFX_FAMILIAR_C2PA_CERT_PEM_PATH;
      const keyPathRaw = process.env.VFX_FAMILIAR_C2PA_PRIVATE_KEY_PEM_PATH;

      // Both unset → signing disabled (D-CTX-2 graceful degradation).
      if (!certPathRaw && !keyPathRaw) return null;

      // Exactly one set → misconfig.
      if (!certPathRaw || !keyPathRaw) {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          'Both VFX_FAMILIAR_C2PA_CERT_PEM_PATH and VFX_FAMILIAR_C2PA_PRIVATE_KEY_PEM_PATH must be set together, or both unset',
          'Set both paths to enable C2PA signing, or unset both to run with signing disabled.',
        );
      }

      // Resolve allowlist root.
      const rootRaw = process.env.VFX_FAMILIAR_C2PA_CERT_ROOT ?? process.cwd();
      let root: string;
      try {
        root = realpathSync(rootRaw);
      } catch {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          'C2PA cert allowlist root does not exist or is not accessible',
          DEFAULT_C2PA_CERT_ROOT_HINT,
        );
      }

      // Resolve + validate each path.
      const certPath = resolveAndValidate(certPathRaw, root, 'cert');
      const keyPath = resolveAndValidate(keyPathRaw, root, 'key');

      // T-14-04 mitigation: warn if key file is world-readable. Concern #4: log
      // basename only.
      const keyStat = statSync(keyPath);
      const mode = keyStat.mode & 0o777;
      if (mode & 0o077) {
        console.error(
          `vfx-familiar: WARNING — C2PA private key file ${basename(keyPath)} has permissive mode 0${mode.toString(8)}; tighten to 0600 for production.`,
        );
      }

      return { certPemPath: certPath, privateKeyPemPath: keyPath };
    }

    function resolveAndValidate(rawPath: string, root: string, label: 'cert' | 'key'): string {
      // Resolve symlinks. realpathSync also asserts existence.
      let resolved: string;
      try {
        resolved = realpathSync(rawPath);
      } catch {
        // Either path doesn't exist OR symlink target doesn't exist.
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA ${label} PEM not readable (file ${basename(rawPath)} not found or symlink broken)`,
          'Ensure the file exists and the server process has read permission. Run scripts/gen-dev-c2pa-cert.mts for a local dev cert.',
        );
      }

      // Allowlist containment. Concern #4: check resolved (post-realpath) path.
      if (resolved !== root && !resolved.startsWith(root + sep)) {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA ${label} path is outside the allowed cert root`,
          `${label} file ${basename(resolved)} resolves outside the allowlist root. ${DEFAULT_C2PA_CERT_ROOT_HINT}`,
        );
      }

      // R_OK check (realpathSync passed, but mode could still deny read).
      try {
        accessSync(resolved, fsConstants.R_OK);
      } catch {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA ${label} PEM not readable: ${basename(resolved)}`,
          'Ensure the server process has read permission on the file.',
        );
      }

      const stat = statSync(resolved);
      if (stat.size === 0) {
        throw new TypedError(
          'C2PA_CONFIG_INVALID',
          `C2PA ${label} PEM is empty: ${basename(resolved)}`,
          `The ${label} file exists but is zero bytes. Regenerate or provide a real PEM.`,
        );
      }

      return resolved;
    }
    ```

    ```typescript
    #!/usr/bin/env tsx
    /**
     * Phase 14 — PROV-V-01. Local dev C2PA cert generator.
     *
     * DEV-ONLY. Generates a self-signed ES256 cert + private key pair to
     * .c2pa-dev/cert.pem + .c2pa-dev/key.pem for testing the C2PA signing
     * path end-to-end without requiring a CA-issued cert. The .c2pa-dev/
     * directory is gitignored so the key never enters git history.
     *
     * Production deployments must use a CA-issued cert (or at minimum a
     * cert whose subject the user trusts). C2PA verifiers will flag this
     * cert as untrusted — that is correct: it is a dev-only fixture.
     *
     * Usage: `npx tsx scripts/gen-dev-c2pa-cert.mts`
     */
    import { generateKeyPairSync } from 'node:crypto';
    import { mkdirSync, writeFileSync, chmodSync } from 'node:fs';
    import { resolve } from 'node:path';
    import { spawnSync } from 'node:child_process';

    const OUT_DIR = '.c2pa-dev';
    mkdirSync(OUT_DIR, { recursive: true });

    // ES256 (P-256 ECDSA) — c2pa-node default per Context7 docs.
    const { privateKey } = generateKeyPairSync('ec', { namedCurve: 'P-256' });
    const keyPath = resolve(OUT_DIR, 'key.pem');
    writeFileSync(keyPath, privateKey.export({ type: 'pkcs8', format: 'pem' }), { mode: 0o600 });
    chmodSync(keyPath, 0o600);

    // Self-signed cert via openssl shell-out (macOS ships with openssl;
    // Linux/CI typically have it via apt/brew). On hosts without openssl,
    // surface a clear error and link to manual cert generation docs.
    const certPath = resolve(OUT_DIR, 'cert.pem');
    const opensslAvail = spawnSync('which', ['openssl'], { stdio: 'ignore' }).status === 0;
    if (!opensslAvail) {
      console.error('vfx-familiar: openssl not found. Install openssl OR generate a cert manually and place at .c2pa-dev/cert.pem.');
      process.exit(2);
    }
    const result = spawnSync('openssl', [
      'req', '-x509', '-key', keyPath, '-out', certPath,
      '-days', '365', '-subj', '/CN=vfx-familiar dev/O=local',
      '-sha256',
    ], { stdio: 'inherit' });
    if (result.status !== 0) {
      console.error('vfx-familiar: openssl req failed; see output above.');
      process.exit(result.status ?? 1);
    }

    console.log(`Generated ${certPath} (mode 0644) and ${keyPath} (mode 0600)`);
    console.log(`Set env: VFX_FAMILIAR_C2PA_CERT_PEM_PATH=${certPath}`);
    console.log(`         VFX_FAMILIAR_C2PA_PRIVATE_KEY_PEM_PATH=${keyPath}`);
    ```
