<objective>
Build the engine-layer C2PA module under `src/engine/c2pa/`. Three focused submodules: a pure manifest-definition builder, a pure format router, and a thin signer wrapper. The signer wrapper is the ONE place that touches `c2pa-node` directly — every downstream consumer (Plan 14-03's engine integration, Plan 14-04's HTTP route, Plan 14-05's verification tests) goes through this module.

This revision applies four BLOCKER/HIGH fixes from the plan-checker:
1. **Concern #1 — algorithm detection.** loadSigner inspects the cert's signature algorithm + public key type via Node X509Certificate (`signatureAlgorithm`, `publicKey.asymmetricKeyType`, `publicKey.asymmetricKeyDetails`) and selects the matching SigningAlgorithm enum value. RSA certs map to PS256/PS384/PS512 (or RS256 if exposed). EC certs map to ES256/ES384/ES512. Ed25519 maps to Ed25519. Unsupported certs throw `C2PA_SIGNER_LOAD_FAILED` with a clear "unsupported algorithm" message — never silently produce invalid signatures.
2. **Concern #10 — RFC4514-safe subject parser.** Replace the naive regex with Node's X509Certificate built-in subject parsing (X509Certificate exposes `subject` as a multi-line DN; we split on newlines + parse each `attr=value` pair, handling RFC4514 escapes for commas, equals, plus, semicolon, less/greater-than).
3. **Concern #11 — native binding load resilience.** Wrap the c2pa-node import in dynamic-import + try/catch; expose `isC2paNodeAvailable()` for downstream guards. loadSigner fails with `C2PA_SIGNER_LOAD_FAILED` when the binding is unavailable rather than crashing the process.
4. **Concern #2 (partial) — sidecar scope reduction.** The format router gains an explicit `unsupported(reason: 'native-handler-missing' | 'unknown-extension')` mode for EXR/PSD (no c2pa-rs handler) and unknown extensions. There is NO `mode: 'sidecar'` in v1.1 — Plan 14-03's engine hook surfaces these as `manifest_signed: false, status_reason: 'unsupported_format'` events. See Plan 14-03 for the full v1.1 / v1.2 scope split rationale.

Purpose: Lock the c2pa-node API surface behind a typed engine-layer facade so Phase 15 (ingredient graph) and Phase 16 (redaction + agent surface) can extend the manifest definition without retouching c2pa-node call sites scattered across the codebase. Architecture-purity preservation is non-negotiable.

Output: Three submodules + tests, architecture-purity guard extended with file-level + directory-level assertions for `src/engine/c2pa/`. No HTTP-route or engine-pipeline integration here — that's Plan 14-03 / 14-04.
</objective>

     * asset.mimeType field.
     */

      | { mode: 'embed-buffer'; mimeType: string }
      | { mode: 'embed-file'; mimeType: string }
      | { mode: 'unsupported'; reason: 'native-handler-missing' | 'unknown-extension'; mimeType?: string };

    /** c2pa-node v0.5.x buffer-API formats (jpeg + png only — verified
     *  bindings.js line 132-134). */
    const BUFFER_TABLE: Record<string, string> = {
      '.png': 'image/png',
      '.jpg': 'image/jpeg',
      '.jpeg': 'image/jpeg',
    };

     *  c2pa-node's buffer API rejects them. Verified against c2pa-rs asset
     *  handlers: BMFF (mp4), RIFF (webp), tiff_io (tif/tiff). */
    const FILE_TABLE: Record<string, string> = {
      '.mp4': 'video/mp4',
      '.webp': 'image/webp',
      '.tif': 'image/tiff',
      '.tiff': 'image/tiff',
    };

    /** c2pa-rs has NO native handler in v0.5.26. v1.1 surfaces these as
     *  unsupported(native-handler-missing). The mimeType is informational
     *  (recorded in the manifest_signed provenance event). v1.2 may add
     *  cryptographic sidecar support if c2pa-node exposes signEmbeddable. */

      '.exr': 'image/x-exr',
      '.psd': 'image/vnd.adobe.photoshop',
    };

    export const EMBED_BUFFER_FORMATS: readonly string[] = Object.values(BUFFER_TABLE);
    export const EMBED_FILE_FORMATS: readonly string[] = Object.values(FILE_TABLE);
    export const UNSUPPORTED_NATIVE_FORMATS: readonly string[] = Object.values(NATIVE_HANDLER_MISSING_TABLE);

    export function routeFormat(filename: string): FormatRoute {
      const dot = filename.lastIndexOf('.');
      if (dot < 0) return { mode: 'unsupported', reason: 'unknown-extension' };
      const ext = filename.slice(dot).toLowerCase();
      if (ext in BUFFER_TABLE) return { mode: 'embed-buffer', mimeType: BUFFER_TABLE[ext]! };

      }
      return { mode: 'unsupported', reason: 'unknown-extension' };
    }

    function detectSigningAlgorithm(
      cert: X509Certificate,
      enumRef: typeof import('c2pa-node').SigningAlgorithm,
    ): import('c2pa-node').SigningAlgorithm {
      const keyType = cert.publicKey.asymmetricKeyType;       // 'ec' | 'rsa' | 'rsa-pss' | 'ed25519'
      const sigAlg = cert.signatureAlgorithm ?? '';            // 'ecdsa-with-SHA256' | 'sha256WithRSAEncryption' | etc.
      const details = cert.publicKey.asymmetricKeyDetails ?? {};

      if (keyType === 'ed25519') return enumRef.Ed25519;

      if (keyType === 'ec') {
        const curve = (details as { namedCurve?: string }).namedCurve ?? '';
        if (curve === 'prime256v1' || curve === 'P-256') return enumRef.ES256;
        if (curve === 'secp384r1' || curve === 'P-384') return enumRef.ES384;
        if (curve === 'secp521r1' || curve === 'P-521') return enumRef.ES512;
        throw new TypedError('C2PA_SIGNER_LOAD_FAILED', `Unsupported EC curve '${curve}' for c2pa-node v0.5.26 (supported: P-256, P-384, P-521)`, 'Reissue the cert with a supported curve.');
      }

      if (keyType === 'rsa-pss' || keyType === 'rsa') {
        const lower = sigAlg.toLowerCase();
        if (lower.includes('sha256')) return enumRef.PS256;
        if (lower.includes('sha384')) return enumRef.PS384;
        if (lower.includes('sha512')) return enumRef.PS512;
        throw new TypedError('C2PA_SIGNER_LOAD_FAILED', `Unsupported RSA signature algorithm '${sigAlg}' for c2pa-node v0.5.26 (supported: SHA-256/384/512 with RSA-PSS)`, 'Reissue the cert with RSA-PSS + SHA-256/384/512.');
      }

      throw new TypedError('C2PA_SIGNER_LOAD_FAILED', `Unsupported public key type '${keyType}' for c2pa-node v0.5.26 (supported: ec, rsa, rsa-pss, ed25519)`, 'Reissue the cert with a supported key type.');
    }

    /** Concern #10 — RFC4514-safe subject parser. */
    function deriveCertSubjectSummary(cert: X509Certificate): string {
      const subject = cert.subject ?? '';
      // X509Certificate.subject is newline-separated 'attr=value' (RFC2253-ish).
      const lines = subject.split(/\r?\n/).filter((l) => l.trim().length > 0);
      const attrs: Record<string, string> = {};
      for (const line of lines) {
        const eqIdx = findUnescapedEquals(line);
        if (eqIdx < 0) continue;
        const attr = line.slice(0, eqIdx).trim();
        const value = unescapeRfc4514(line.slice(eqIdx + 1).trim());
        if (!(attr in attrs) && value.length > 0) attrs[attr] = value;
      }
      if (attrs['CN']) return attrs['CN'];
      if (attrs['O']) return attrs['O'];
      const fp = cert.fingerprint256.replace(/:/g, '').toLowerCase();
      return `fp:${fp.slice(0, 16)}`;
    }

    function findUnescapedEquals(s: string): number {
      for (let i = 0; i < s.length; i++) {
        if (s[i] === '=' && (i === 0 || s[i - 1] !== '\\')) return i;
      }
      return -1;
    }

    function unescapeRfc4514(s: string): string {
      return s.replace(/\\([,=+;<>"\\])/g, '$1');
    }

          `c2pa-node sign() rejected the asset: ${(err as Error).message}`,
